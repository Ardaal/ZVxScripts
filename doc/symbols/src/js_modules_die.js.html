 22 ({
 23 
 24      require: ["commands", "com", "theme", "user", "io"],
 25 
 26 
 27      loadModule: function ()
 28      {
 29          this.commands.registerCommand("die", this);
 30          this.io.registerConfig(this, {waysOfDying: [
 31                                            "% found an anvil from above.", "% got hit by a meteor.", "Salamence thought % was tasty.",
 32                                            "% looked a bit too much like fish food to Sharpedo.",
 33                                            "% went dancing with Blaziken.", "% listened to Lapras sing.", "% was tackled by Magikarp!"
 34                                        ]
 35                                       });
 36      },
 37 
 38 
 39      die:
 40      {
 41          server: true,
 42 
 43          desc: "Leave the server with style.",
 44 
 45          perm: function ()
 46          {
 47              return true;
 48          },
 49 
 50          code: function (src, cmd, chan)
 51          {
 52              var i = Math.floor(Math.random() * this.config.waysOfDying.length);
 53 
 54              this.com.broadcast(this.config.waysOfDying[i].replace(/%/g, this.user.name(src)), this.theme.GAME);
 55 
 56              if (src) sys.kick(src);
 57          }
 58 
 59      }
 60 
 61 });
