 28 /*
 29 
 30 RPG is per-channel
 31 
 32 One global RPG object per channel.
 33 
 34 PLayers will have skills, some for combat, some passive, some field.
 35 
 36 Some important field skills:
 37 
 38 Rest: Recovers the players statuses.
 39 Dig: Digs up things.
 40 Craft: Combine materials to make things.
 41 Meditate: Increases passive skill experience
 42 Train: Increase active skill experience.
 43 
 44 Passive skills example:
 45 
 46 Magical Box: Allows you to bring far more items than you could possibly carry.
 47 Aura Shield: Reduces damage from magical attacks.
 48 Sharp Vision: Sees things
 49 Sharp Hearing: Hears things
 50 Sharp Sense: ...
 51 
 52 
 53 Material Properties:
 54 
 55 hardness     - Harder materials are harder to craft, they also make better blades.
 56 density      - Dense materials reduce the player's speed more.
 57 resistance - Related to electrical attacks.
 58 thermal      - How much the item resists heat and cold
 59 opacity      - Related to laser attacks
 60 
 61 
 62 Entity properties:
 63 
 64 playername: if a player owns this entity, otherwise null
 65 type: class name // e.g., null for player, otherwise e.g.: "wolf", "skydragon"
 66 str: strength points
 67 res: resillience points
 68 spd: speed points
 69 mag: magic points
 70 psy: psychic points
 71 spr: spirit points
 72 
 73 hp: // max hp points determined by sta
 74 sta: // max determined by res
 75 men: // mental stamina, determined by psy
 76 mana: // max mana determined by mag
 77 lp: // max life energy determined by spr
 78 
 79 head: head item
 80 body: body item
 81 feet: foot item
 82 lhand: left hand item
 83 rhand: right hand item
 84 items: storage
 85 
 86 */

118     step: function ()
119     {
120         gametick: for (var x in this.games)
121         {
122             if (this.games[x].paused) continue gametick;
123             if (!this.games.ready) continue gametick;
124             this.games[x].tick++;
125 
126             playerstep: for (var x2 in this.games[x].players)
127             {
128                 if (!sys.id(x2)) 
129                     // player offline
130                 {
131                     continue playerstep;
132                 }
133 
134                 var p = this.games[x].players[x2];
135 
136                 var ctx = 
137                     {
138                         game: this.games[x],
139                     };
140 
141                 if (!player.inbattle) this.idleStep(p, ctx);                    
142             }
143         }
144     }

174     ,
175     entityTick: function (e, ctx)
176     {
177         for (var x in e.activeSkills)
178         {
179             var s = e.activeSkills[x];
180             if (s.timer) 
181             {
182                 s.timer--;
183                 if (s.timer <= 0) 
184                 {
185                     delete e.activeSkills[x];
186                     continue;
187                 }
188             }
189             
190             e.hp += s.hp || 0;
191             e.sta += s.sta || 0;
192             e.mana += s.mana || 0;
193             e.mst += s.mst || 0;
194 
195             if (s.effect) this.effects[s.effect](e, ctx);
196         }
197 
198         if (e.hp <= 0) this.entityDie(e);
199         if (e.sta <= 0) this.entityPurge(e);
200         if (e.mana <= 0) this.entityPurge(e);
201         if (e.mst <= 0) this.entityPurge(e);
202       
203     }
204     ,
205     equipPhysAtk: function (equip)
206     {
207         if (equip === null) return;
208         var b = this.equips[equip.type].base;
209         var a = this.materials[equip.type].amount;
210         var s = this.materials[equip.material].sharpness;
211         var d = this.materials[equip.material].density;
212 
213 
214         var val = b * s * Math.log(d*a) / 100; 
215         // e.g., Iron Sword: 40 * 40 * ~4.382 / 100 = ~700
216         // Diamond Broadsword: 200 * 500 * ~7.937 / 100= ~8000
217  
218         var q = equip.quality;
219 
220         var factor = 1 - ( 1 / Math.log(q + Math.E) ); // e.g, quality of 
221 
222         return val *factor;
223     }

231         }
232     }

242     entityPhsyAtk: function (entity)
243     {
244         return Math.log(entity.str + Math.E + this.entityModPhys(entity)) * (this.equipPhysAtk(entity.lhand) + this.equipPhysAtk(entity.rhand));
245     }

247     entityPhysDef: function ()
248     {
249         return Math.log(this.entityStr(entity) + Math.E) * (this.equipPhysAtk(entity.lhand) + this.equipPhysAtk(entity.rhand));
250     }

252     entityMaxHP: function (e)
253     {
254         return (e.str*0.1 + e.res*0.3 + (Math.log(e.res/1000+Math.E)*15000 | 0));
255     }
256     ,
257     entityMaxMP: function (e)
258     {
259         return (e.mag*1.2 + (Math.log(e.res*0.3+Math.E)*70 | 0));
260     }
261     ,
262     entityMaxSP: function (e)
263     {
264         return (e.str*0.2 + e.res*0.2 + (Math.log(e.res/1000+Math.E)*15000 | 0));
265     }

270     }
