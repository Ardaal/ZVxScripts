 98     playerStep: function (player, ctx)
 99     {
100         if (player.battling) return;
101         // Regular player events don't occur while the player is in a battle!
102 
103         for (var x in player.activeActions)
104         {
105             if ("timer" in player.activeActions[x] && player.activeActions[x].timer-- <= 0)
106             {
107                 if (player.activeActions[x].done) this.actions[player.activeActions[x].done].call(this, player.activeActions[x], { rpg: ctx.rpg, player:player, index: x });
108                 delete player.activeActions[x];
109             }
110             else if (player.activeActions[x].tick)
111             {
112                  this.actions[player.activeActions[x].tick].call(this, player.activeActions[x], { rpg: ctx.rpg, player:player, index: x });
113             }
114             break;
115         }
116 
117         this.playerUpdateStats(player);
118     }
119     ,
120 
121     playerUpdateStats: function (e)
122     {
123         e.maxmp =  (e.mag*0.13 + (Math.log(e.mag+Math.E)*10 | 0));
124         e.maxsp = (e.str*0.02 + e.res*0.02 + (Math.log(e.res/100+Math.E)*150 | 0));
125         e.maxmsp = (e.res*0.01 + e.mag*0.01 + e.psy*0.12 + e.spr*0.01 + (Math.log(e.psy/1000+Math.E)*50 | 0));
126         e.maxhp = (e.str*0.01 + e.res*0.03 + (Math.log(e.res/100+Math.E)*100 | 0));
127         e.power = Math.floor(800*Math.log(3/2*e.str+e.psy/3+Math.E)+e.psy/25000+300);
128         e.offense = Math.floor(e.power / 10000 * (100 + this.equipAtk(e.lhand) + this.equipAtk(e.rhand)));
129         e.defense = Math.floor(e.power / 10000 * (100 + this.equipDef(e.lhand)/2 + this.equipDef(e.rhand)/2 + this.equipDef(e.body) + this.equipDef(e.feet) + this.equipDef(e.head) + this.equipDef(e.back)));
130     }
