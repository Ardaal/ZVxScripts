 22 ({
 23     require: ["com", "theme", "chat", "util", "commands"]
 24     ,
 25     loadModule: function ()
 26     {
 27         this.commands.registerCommand("help", this);
 28     }
 29     ,
 30     help:
 31     {
 32         perm: function (src)
 33         {
 34             return true;
 35         }
 36         ,
 37         code: function (src, cmd, chan)
 38         {
 39             this.com.message([src], "What do you need help with?<br/>Using commands, type 1.<br/>Contacting server admins, type 2.<br/>Getting source code, type 3.<br/>Anything else, type 4.<br/>To cancel this and go back to the chat, type exit.</span>", this.theme.INFO, true);
 40 
 41             this.chat.registerCapture(src, this.helper_capturer, this);
 42         }
 43     }
 44     ,
 45     helper_capturer: function (src, msg, chan)
 46     {
 47         var emsg = msg.replace(/[\s\.\,\?\!]/g, "");
 48         if (emsg === "exit") return;
 49         
 50         if (emsg === "1") 
 51         {
 52             this.com.message(
 53                 [src]
 54                 ,
 55                 "Commands can be used similar to optargs style, for example: " +
 56                     "<b><font color=grey>/</font><font color=blue>command</font> <font color=red>\"</font><font color=teal>name of player</font><font color=red>\"</font> "+
 57                     "<font color=green>--</font><font color=orange>reason</font><font color=purple>=</font>"+
 58                     "<font color=red>\"</font><font color=gold>reason goes here</font><font color=red>\"</font></b><br/>" +
 59                     "The grey <b>/</b> is the slash used to indicate you are typing a command.<br/>"+
 60                     "The blue <b>command</b> is the name of the command to use.<br/>"+
 61                     "The red <b>\"</b> are quotation marks, these show where a parameter begins and ends.<br/>"+
 62                     "The teal <b>name of player</b> is a standard parameter.<br/>"+
 63                     "The green <b>--</b> indicates the start of a <i>named flag</i>.<br/>"+
 64                     "The orange <b>reason</b> is the name of a named flag.<br/>"+
 65                     "The purple <b>=</b> is an assignment, associating a named flag with a parameter.<br/>"+
 66                     "The gold <b>reason goes here</b> is a parameter to the --reason named flag.<br/>"
 67                 ,
 68                 this.theme.INFO, true
 69             );
 70 
 71             this.com.message([src], "Type /cmdlist for a list of commands, include the --all option to list all commands.", this.theme.INFO, true);
 72             this.com.message([src], "What do you need help with?<br/>Using commands, type 1.<br/>Contacting server admins, type 2.<br/>Getting source code, type 3.<br/>Anything else, type 4.<br/>To cancel this and go back to the chat, type exit.</span>", this.theme.INFO, true);
 73             this.chat.registerCapture(src, "helper_capturer", this);
 74             return;
 75         }
 76 
 77         else if (emsg === "2")
 78         {
 79             this.com.message([src], "Admins have special pokeballs, look for great ball or higher!", this.theme.info);
 80             this.com.message([src], "What do you need help with?<br/>Using commands, type 1.<br/>Contacting server admins, type 2.<br/>Getting source code, type 3.<br/>Anything else, type 4.<br/>To cancel this and go back to the chat, type exit.</span>", this.theme.INFO, true);
 81             this.chat.registerCapture(src, "helper_capturer", this);
 82             return;
 83         }
 84 
 85         else if (emsg === "3")
 86         {
 87             this.script.AGPL(src, msg, chan);
 88             this.com.message([src], "What do you need help with?<br/>Using commands, type 1.<br/>Contacting server admins, type 2.<br/>Getting source code, type 3.<br/>Anything else, type 4.<br/>To cancel this and go back to the chat, type exit.</span>", this.theme.INFO, true);
 89             this.chat.registerCapture(src, "helper_capturer", this);
 90             return;
 91         }
 92 
 93         else if (emsg === "4")
 94         {
 95             this.com.message([src], "Go ask someone in the chat!", this.theme.INFO);
 96             return;
 97         }
 98 
 99         this.com.message([src], "Command not understood.", this.theme.INFO);
100         this.com.message([src], "What do you need help with?<br/>Using commands, type 1.<br/>Contacting server admins, type 2.<br/>Getting source code, type 3.<br/>Anything else, type 4.<br/>To cancel this and go back to the chat, type exit.</span>", this.theme.INFO, true);
101         this.chat.registerCapture(src, "helper_capturer", this);
102    }
103 });
