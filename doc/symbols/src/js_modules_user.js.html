344      nameMajorGroups: function(name)
345      {
346          var ugroups, groups, majors;
347          var lname = name.toLowerCase();
348 
349          if (lname == "~~server~~")
350          {
351              return {"Sever":null};
352          }
353 
354          if (sys.dbRegistered(lname))
355          {
356              groups = {"Registered": null};
357          }
358          else return {"User": null};
359 
360          if (!this.relMajors[lname]) ugroups = [];
361          else ugroups = this.relMajors[lname];
362 
363          if (ugroups) for (var x in ugroups)
364          {
365              groups[ugroups[x]] = null;
366 
367          }
368 
369          if ((!("User" in groups))  &&  (!("Moderator" in groups))  &&  (!("Administrator" in groups))  &&  (!("Owner" in groups)))
370          {
371              groups.User = null;
372          }
373 
374 
375          return groups;
376      },

379      beforeLogIn: function (src)
380      {
381          var groups = this.majorGroups(src);
382 
383          if ( src != 0 &&  ((!("User" in groups))  &&  (!("Moderator" in groups))  &&  (!("Administrator" in groups))  &&  (!("Owner" in groups)) || this.auth(src) != 0 ) )
384          {
385              var a = this.auth(src);
386 
387              if (a)
388              {
389                  this.majorGroupAddMember([null, "Moderator", "Administrator", "Owner"][this.auth(src)], src);
390              }
391          }
392      },

405 
406          var group = this.database.majorgroupinfo[groupname];
407

531      changeInvisible: function (id, bool)
532      {
533          if (!id) return;
534 
535          var lname = this.name(id).toLowerCase();
536          if (lname in this.invisibleUsers && !bool)
537          {
538              sys.changeAuth(id, this.invisibleUsers[lname]);
539              delete this.invisibleUsers[lname];
540          }
541 
542          if (bool && !(lname in this.invisibleUsers))
543          {
544              this.invisibleUsers[lname] = this.auth(id);
545              sys.changeAuth(id, 0);
546          }
