 22 ({
 23 
 24     require: ["com", "commands", "theme", "util", "less"]
 25     ,
 26     eval:
 27     {
 28         desc: "Executes code on the server"
 29         ,
 30         perm: function (src)
 31         {
 32             return sys.auth(src) == 3;
 33         }
 34         ,
 35         code: function (src, cmd, chan)
 36         {
 37             try 
 38             {
 39                 this.less.less(src, this.util.inspect(eval(cmd.input), true), false);
 40             } 
 41             catch (e)
 42             {
 43                 this.com.message([src], e.toString(), this.theme.CRITICAL);
 44                 this.com.message([src], e.backtracetext, -1);
 45             }
 46         }
 47     }
 48     ,
 49     loadModule: function ()
 50     {
 51         this.commands.registerCommand("eval", this);
 52     }
 53 });
