 22 ({
 23     require: ["commands", "com", "theme"]
 24     ,
 25     setauth:
 26     {
 27         aliases: ["auth"]
 28         ,
 29         desc: "Set user auth level"
 30         ,
 31         options:
 32         {
 33             level: "What level to set the user to"
 34         }
 35         ,
 36         perm: function (src)
 37         {
 38             return sys.auth(src) == 3;
 39         }
 40         ,
 41         code: function (src, cmd, chan)
 42         {
 43             var levels = 
 44                 {
 45                     "3":3, "owner":3, "root":3,
 46                     "2":2, "admin":2, "administrator":2,
 47                     "1":1, "mod":1, "moderator":1,
 48                     "0":0, "user": 0, "none":0
 49                 };
 50             var level = null;
 51 
 52 
 53             if (cmd.flags.level.toLowerCase() in levels)
 54             {
 55                 level = levels[cmd.flags.level.toLowerCase()];
 56             }
 57 
 58             else
 59             {
 60                 this.com.message([src], "Unknown level", this.theme.WARN);
 61                 return;
 62             }
 63 
 64             for (var x in cmd.args)
 65             {
 66                 if (! sys.dbIp(cmd.args[x]))
 67                 {
 68                     this.com.message([src], "User unknown", this.theme.WARN);
 69                     continue;
 70                 }
 71                 if (! sys.dbRegistered(cmd.args[x])) 
 72                 {
 73                     this.com.message([src], "User unregistered", this.theme.WARN);
 74                     continue;
 75                 }
 76                 var i = sys.id(cmd.args[x]);
 77 
 78                 if (i) 
 79                 {
 80                      sys.changeAuth(i, level);
 81                 }
 82                 else
 83                 {
 84                     sys.changeDbAuth(cmd.args[x], level);
 85                 }
 86 
 87                 this.com.broadcast(sys.name(src) + " set " + cmd.args[x] + " to level " + level, this.theme.INFO);
 88             }
 89         }
 90     }
 91     ,
 92     loadModule: function()
 93     {
 94         this.commands.registerCommand("setauth", this);
 95     }
 96 });
