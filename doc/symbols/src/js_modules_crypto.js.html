 24      Locker: function (password, object)
 25      {
 26          var sha = sys.sha1;
 27 
 28          var salt = Math.random();
 29          var cryptpass = sha(  salt.toString() + password );
 30 
 31          password = null;
 32 
 33          return (
 34              function (passwd)
 35              {
 36                  if (sha(salt.toString() + passwd) == cryptpass) return object;
 37 
 38                  else return null;
 39              }
 40          );

 43      HardCrypt: function (key, data)
 44      {
 45          // Combination of a stream and block cipher
 46          var output, hash, salt, crypto, cipher, passes, temp, ciphers, i, subcipher, i2, unencrypted_output, compressed_output, num_blocks;
 47 
 48          if (typeof data == "string") data = sys.ByteArray(data);
 49          if (typeof key == "string") key = sys.ByteArray(key);
 50 
 51          salt = sys.ByteArray(8);
 52 
 53          for (i = 0; i < 8; i++)
 54          {
 55              salt[i] = Math.floor(Math.random() * 256);
 56          }
 57 
 58          cipher = this.sha1x128(sys.ByteArray(0).append(salt).append(key));
 59 
 60          output = sys.ByteArray(0).append(salt);
 61 
 62          unencrypted_output = sys.ByteArray(8);
 63 
 64 
 65          // Add garbage to make the data more secure
 66          for (i = 0; i < 8; i++)
 67          {
 68              unencrypted_output[i] = Math.floor(Math.random() * 256);
 69          }
 70 
 71          unecrypted_output.append(data.length); // Append data's length
 72 
 73          unencrypted_output.append(data); // Add the data
 74 
 75          unencrypted_output.append(sys.sha1Binary(unencrypted_output)); // Add hash to prevent tampering.
 76 
 77          i2 = Math.floor(Math.random()*200); // Arbitrary amount of garbage
 78 
 79          temp = sys.ByteArray(i2);
 80 
 81          // Add garbage to make the data more secure
 82          for (i = 0; i < i2; i++)
 83          {
 84              temp[i] = Math.floor(Math.random() * 256);
 85          }
 86 
 87          unencrypted_output.append(temp);
 88 
 89          // final unecrypted_output
 90 
 91          compressed_output = sys.qCompress(unencrypted_output);
 92 
 93          // now add garbage so that compressed output is of block size:
 94 
 95          while(compressed_output.length % 64 != 0)
 96          {
 97              temp = sys.ByteArray(1);
 98              temp[0] = Math.floor(Math.random() * 256);
 99              compressed_output.append(temp);
100          }
101 
102          num_blocks = compressed_output.length / 64;
103 
104          ciphers = [];
105 
106          for (i = 0; i < num_blocks; i++)
107          {
108              ciphers[i] = this.applyCipher(cipher, (i == 0? cipher: ciphers[i - 1]));

114      sha1x128: function (data)
115      {
116          var output, i;
117          output = sys.ByteArray(0);
118 
119          for (i = 0; i < 7; i++)
120          {
121              output.append( sys.sha1Binary( data.append(this.intToBin(i)) ));
122          }
123 
124          output.truncate(128);
125 
126          return output;
127      },
