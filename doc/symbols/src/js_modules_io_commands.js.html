 22 ({
 23     require: ["io", "com", "logs", "theme", "commands", "less", "user"]
 24     ,
 25     loadModule: function ()
 26     {
 27         this.commands.registerCommand("ioctrl", this);
 28     }
 29     ,
 30     ioctrl:
 31     {
 32         server: true
 33         ,
 34         desc: "Controls I/O Layer"
 35         ,
 36         options:
 37         {
 38             purge: "Delete database(s)"
 39             ,
 40             sync: "Write database(s) to file"
 41             ,
 42             backup: "Back-up databases(s)"
 43             ,
 44             close: "Closes databases"
 45             ,
 46             all: "Apply actions to all open databases as well"
 47         }
 48         ,
 49         perm: function (src)
 50         {
 51             return "IO" in this.user.groups(src);
 52         }
 53         ,
 54         code: function (src, cmd)
 55         {
 56             var dblist = cmd.args;
 57             var now = +new Date;
 58 
 59             if (!cmd.flags.all && !cmd.flags.sync && !cmd.flags.backup && !cmd.flags.purge && cmd.args.length == 0)
 60             {
 61                 this.com.message([src], "Loaded databases:", this.theme.INFO);
 62                 this.less.less(src, Object.keys(this.io.openDBs).join("\n"), false);
 63                 return;
 64             }
 65 
 66 
 67 
 68             if (cmd.flags.all)
 69             {
 70                 var dblist2 = Object.keys(this.io.openDBs);
 71                 for (var x in dblist2) if (dblist.indexOf(dblist2[x]) === -1) dblist.push(dblist2[x]);
 72             }
 73 
 74             if (cmd.flags.sync) for (var x in dblist)
 75             {
 76                 var start = +new Date;
 77                 this.io.flushDB(dblist[x]);
 78                 var end = +new Date;
 79 
 80             }
 81 
 82             if (cmd.flags.commit) for (var x in dblist)
 83             {
 84                 var start = +new Date;
 85                 this.io.commitDB(dblist[x]);
 86                 var end = +new Date;
 87 
 88             }
 89 
 90             if (cmd.flags.backup) for (var x in dblist)
 91             {
 92                 var start = +new Date;
 93                 this.io.backupDB(dblist[x]);
 94                 var end = +new Date;
 95 
 96             }
 97 
 98             if (cmd.flags.close) for (var x in dblist)
 99             {
100                 var start = +new Date;
101                 this.io.closeDB(dblist[x]);
102                 var end = +new Date;
103 
104             }
105 
106             if (cmd.flags.purge)
107             {
108                 for (var x in dblist)
109                 {
110                     this.io.purgeDB(dblist[x]);
111 
112                 }
113                 this.script.reloadModule("io");
114             }
115         }
116     }
117 });
