 85     beforeBattleStarted: function (source, target, clauses, rated, mode, battleid, team1, team2)
 86     {
 87         if (!this.tourOf[source] && !this.tourOf[target]) return;
 88

 90 
 91         if (tour != this.tourOf[target])

 95             return;

100         matchup_check:
101         {
102             var matchups = tour.matchups;
103 
104             for (var x in matchups) if (matchups[x].indexOf(this.user.name(source).toLowerCase()) != -1 && matchups[x].indexOf(this.user.name(target).toLowerCase()) !== -1)
105             {
106                 matchup_id = x;
107                 break matchup_check;
108             }
109 
110             this.com.message([source, target], "Players in a tournament may only battle their opponent!", this.theme.WARN);
111             sys.stopEvent();
112             return;
113         }
114 
115         var tier = tour.tier;
116 
117         if (tier)
118         {
119             if (sys.tier(source, team1) !== tier)
120             {
121                 this.com.message([source, target], this.user.name(source) + " is in the wrong tier!");
122                 sys.stopEvent();
123                 return;
124             }
125 
126             if (sys.tier(target, team2) !== tier)
127             {
128                 this.com.message([source, target], this.user.name(target) + " is in the wrong tier!");
129                 sys.stopEvent();
130                 return;
131             }
132         }
133 
134         this.matchIDs[source] =
135         this.matchIDs[target] =
136                                 battleid;
137 
138         return;
