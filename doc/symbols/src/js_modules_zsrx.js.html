 26      /** The ZSrX Function is an faster resource undemanding version of JSON.stringify. It has some limitations, in general, it ignores edge cases.
 27       *
 28       */
 29      zsrx: function (inst)
 30      {
 31          var olist = [];
 32          var strlt = [];
 33 
 34          var depth = 0;
 35          var dstr = "";
 36 
 37          var flashstr = [];
 38 
 39          var a = 3e+2; // anti-hang circular counter
 40 
 41          function srsz (variant)
 42          {
 43              if (typeof variant == "number") strlt.push(""+ variant);
 44 
 45              else if (typeof variant == "string") strlt.push(JSON.stringify(variant));
 46 
 47              else if (variant === null) strlt.push("null");
 48 
 49              else if (typeof variant === "object" && variant instanceof Array)
 50              {
 51                  if (--a === 0)
 52                      // Check for circular references
 53                  {
 54                      // if (olist.indexOf(variant) !== -1) throw new Error("Inconsitent object");
 55                      a = 3e+5;
 56                      flashstr.push(strlt.join(""));
 57                      strlt = [];
 58                      gc();
 59                  }
 60 
 61                  strlt.push("[\n");
 62                  dstr += " "; // indent
 63 
 64                  for (var x in variant)
 65                  {
 66 
 67                      strlt.push(dstr);
 68 
 69                      srsz(variant[x]);
 70 
 71                      strlt.push(",\n");
 72                  }
 73 
 74                  if (strlt[strlt.length-1] === ",\n") strlt.pop(); // remove trailing comma
 75 
 76                  dstr = dstr.substring(1); // unindent
 77 
 78                  strlt.push("\n" +dstr+"]");
 79              }
 80 
 81              else if (typeof variant == "object")
 82              {
 83                  if (--a === 0)
 84                      // Check for circular references
 85                  {
 86                      //if (olist.indexOf(variant) !== -1) throw new Error("Inconsitent object");
 87 
 88                      a = 3e+4;
 89                      flashstr.push(strlt.join(""));
 90 
 91                      strlt = [];
 92                      gc();
 93                  }
 94 
 95                  strlt.push("{\n");
 96                  dstr += " "; // indent
 97 
 98                  for (var x in variant)
 99                  {
100                      if (variant[x] === undefined || typeof variant[x] == "function") continue;
101 
102                      strlt.push(dstr + JSON.stringify(x) + ": ");
103 
104                      srsz(variant[x]);
105 
106                      strlt.push(",\n");
107                  }
108 
109                  if (strlt[strlt.length-1] === ",\n") strlt.pop(); // remove trailing comma
110 
111                  dstr = dstr.substring(1); // unindent
112 
113                  strlt.push("\n"+dstr+"}");
114              }
115 
116              else if (typeof variant === "boolean") strlt.push(""+variant);
117 
118              else strlt.push("null");
119 
120 
121 
122 
123          }
124 
125 
126 
127          srsz(inst);
128 
129          if (strlt[strlt.length-1] == "]\n")
130          {
131              strlt.pop();
132              strlt.push("]");
133          }
134 
135          if (strlt[strlt.length-1] == "}\n")
136          {
137              strlt.pop();
138              strlt.push("}");
139          }
140 
141 
142          return flashstr.join("") + strlt.join("");
143      }
