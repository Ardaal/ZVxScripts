1 ({ // starts module
  2     require: ["commands", "com", "io"],
  3     loadModule: // rules.loadModule
  4     function ()
  5     {
  6         this.commands.registerCommand("rules", this);
  7 
  8         this.io.registerConfig(this, { rules: ["No trolling.", "No flaming.", "No spamming."], sendRulesOnLogin: 0 });
  9 
 10         this.script.registerHandler("afterLogIn", this);
 11     },
 12 
 13 
 14     rules:
 15     {
 16         perm: function ()
 17         {
 18             return true; // all users should be able to use /rules so don't check anything
 19         },
 20 
 21         code: function (src, cmd, chan) // code that runs /rules
 22         {
 23              var x; // declare variable x
 24 
 25              for (x in this.config.rules)
 26              {
 27                  this.com.message(src, "Rule #" + x + ": " + this.config.rules[x]);
 28              }
 29         }
 30 
 31     },
 32 
 33     afterLogIn: function (src)
 34     {
 35         if (this.config.sendRulesOnLogin == 1)
 36         {
 37             this.com.message(src, "Please type /rules to read the rules!");
 38         }
 39 
 40         if (this.config.sendRulesOnLogin == 2)
 41         {
 42             var x; // declare variable x
 43 
 44             for (x in this.config.rules)
 45             {
 46                 this.com.message(src, "Rule #" + x + ": " + this.config.rules[x]);
 47             }
 48         }
 49     }
 50 });
