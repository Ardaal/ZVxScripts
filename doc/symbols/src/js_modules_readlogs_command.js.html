 22 ({
 23     require: ["com", "theme", "logs", "commands", "less", "text", "user"]
 24     ,
 25     readlogs:
 26     {
 27         server: true,
 28 
 29         options:
 30         {
 31             count: "How many entries to read.",
 32             trace: "Show backtraces if available.",
 33             types: "Comma separated types of log messages to show",
 34             match: "Pattern matching"
 35         },
 36 
 37         perm: function (src)
 38         {
 39             return "LOGS" in this.user.groups(src);
 40         }
 41         ,
 42         code: function (src, cmd, chan)
 43         {
 44             var g = this.user.groups(src);
 45 
 46             var count = (+cmd.flags.count) || 35;
 47             var ptypes = cmd.flags.types || cmd.flags.type;
 48             if (ptypes) ptypes = ptypes.split(/,/g);
 49             else ptypes = ["chat", "script", "io", "scripterror", "user", "command", "security"];
 50             var trace = cmd.flags.trace;
 51             var types = [];
 52 
 53             if (! ("SERVEROP" in g || "LOGS[*]" in g) ) for (var x in ptypes)
 54             {
 55                 if (("LOGS[" + ptypes[x].toUpperCase() + "]") in g)
 56                 {
 57                     types.push(ptypes[x]);
 58                 }
 59 
 60                 else if (cmd.flags.type || cmd.flags.types) // only warn if the user specified types manually.
 61                 {
 62                     this.com.message(src, "Permission Denied: Can't view logs of type " + ptypes[x] + ".", this.theme.WARN);
 63                 }
 64             }
 65 
 66             else types = ptypes;
 67 
 68 
 69             var msgs = [];
 70 
 71             for (var i = this.logs.logs.length - 1; i >= 0 && count > 0; i--)
 72             {
 73                 if (types.indexOf(this.logs.logs[i].level) !== -1 && (!cmd.flags.match || this.text.escapeHTML(this.logs.logs[i].msg).indexOf(cmd.flags.match) !== -1))
 74                 {
 75 
 76                     msgs.push(
 77                         "<b>Type " + this.logs.logs[i].level + "</b> at <em>" + (this.logs.logs[i].time||"?") + "</em>: " + this.text.escapeHTML(this.logs.logs[i].msg) +
 78                             ( !trace || !this.logs.logs[i].trace ? "" : "<br/><code>" + this.text.escapeHTML(this.logs.logs[i].trace) + "</code>")
 79                     );
 80                     count--;
 81                 }
 82             }
 83 
 84             this.less.less(src, "<b>Logs:</b><hr/>" + msgs.join("<br/>") + "<hr/>", true);
 85         }
 86     }
 87     ,
 88     loadModule: function ()
 89     {
 90         this.commands.registerCommand("readlogs", this);
 91     }
 92 });
