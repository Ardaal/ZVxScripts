 22 ({
 23      require: ["text", "time"],
 24      hotswap: true,
 25 
 26      INFO: 0,
 27      WARN: 1,
 28      CRITICAL: 2,
 29      GAME: 3,
 30      TOUR: 4,
 31      LOG: 5,
 32 
 33 
 34      TEMPLATE1: "<span style='color:%1;'><timestamp /><b>~Script~:</b></span> %2",
 35      TEMPLATE2: "<span style='color:%1;'><timestamp /><b>~Script~:</b> %2</span>",
 36 
 37 
 38      formatAs: function (text, type)
 39      {
 40          switch (type)
 41          {
 42          case this.INFO:
 43              return this.TEMPLATE1.replace(/%1/g, "blue").replace(/%2/, text);
 44 
 45          case this.WARN:
 46              return this.TEMPLATE1.replace(/%1/g, "red").replace(/%2/, text);
 47 
 48          case this.CRITICAL:
 49              return this.TEMPLATE2.replace(/%1/g, "red").replace(/%2/, text);
 50 
 51          case this.GAME:
 52              return this.TEMPLATE1.replace(/%1/g, "green").replace(/%2/, text);
 53 
 54          case this.TOUR:
 55              return this.TEMPLATE2.replace(/%1/g, "green").replace(/%2/, text);
 56 
 57          case this.LOG:
 58              return this.TEMPLATE2.replace(/%1/g, "grey").replace(/%2/, text);
 59 
 60 
 61          default:
 62              return text;
 63          }
 64      },
 65 
 66      issuetext: function (issued)
 67      {
 68          var outtext = [];
 69          function out(prop, propname)
 70          {
 71              if (issued[prop] && issued[prop].length)
 72              {
 73                  outtext.push(propname + ": " + issued[prop].join(", ") + ".");
 74              }
 75          }
 76 
 77          if (issued.reason) outtext.push("Reason: " + issued.reason);
 78 
 79          outtext.push("Expires: " + ( issued.expires?  this.time.diffToStr(issued.expires - +new Date) + " (" +  new Date(issued.expires).toString() +")" : "indefinite"));
 80 
 81          out.call(this, "names", "Names");
 82          out.call(this, "ips", "IPs");
 83          out.call(this, "subnets", "IP Subnets");
 84          out.call(this, "nameRegex", "Regular Expressions");
 85          
 86          out.call(this, "hostnames", "Hostname Regular Expression(s)");
 87 
 88          return outtext.join("\n");
 89 
 90      },
 91 
 92 
 93 
 94      issuehtml: function (issued)
 95      {
 96          var outtext = [];
 97          function out(prop, propname)
 98          {
 99              if (issued[prop] && issued[prop].length)
100              {
101                  outtext.push("<b>"+  propname + "</b>: " + this.text.escapeHTML(issued[prop].join(", ")) + ".");
102              }
103          }
104 
105          if (issued.reason) outtext.push("<b>Reason:</b> " + issued.reason);
106 
107          outtext.push("<b>Expires:</b> " + ( issued.expires?  this.time.diffToStr(issued.expires - +new Date) + " (" +  new Date(issued.expires).toString() +")" : "indefinite"));
108 
109          out.call(this, "names", "Name(s)");
110          out.call(this, "ips", "IP(s)");
111          out.call(this, "subnets", "IP Subnet(s)");
112          out.call(this, "nameRegex", "Regular Expression(s)");
113          out.call(this, "hostnames", "Hostname Regular Expression(s)");
114 
115          return outtext.join("<br/>");
116 
117      }
118 
119 
120  });
