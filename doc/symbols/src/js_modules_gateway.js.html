 22 ({
 23      require: ["security",  "com", "theme", "time", "logs", "user", 'util'],
 24 
 25 
 26      loadModule: function()
 27      {
 28          this.script.registerHandler("beforeLogIn", this );
 29          this.script.registerHandler("beforeLogOut", this );
 30          this.script.registerHandler("afterChangeTeam", this);
 31      },
 32 
 33 
 34      beforeLogIn: function(src)
 35      {
 36 
 37 
 38          var banID = this.security.banID(src);
 39 
 40          if (banID)
 41          {
 42 
 43              var ban = this.security.database.bans[banID];
 44 
 45              this.com.message(
 46                  [src],
 47                  "<hr/>You are banned.<br>" + this.theme.issuehtml(ban) + "<hr/>",
 48                  this.theme.CRITICAL,
 49                  true
 50              );
 51 
 52              if (!this.user.hasPerm(src, "PROTECTED"))
 53              {
 54 
 55                  this.logs.logMessage(this.logs.WARN, "Banned user: " + sys.name(src) + " (IP: " + sys.ip(src) + ") (#: " + banID + ") tried to log in.");
 56                  sys.stopEvent();
 57              }
 58              else
 59              {
 60                  this.com.message(src, "Note: Your account is protected, and therefore the ban is void.");
 61              }
 62          }
 63 
 64 
 65 
 66 
 67          this.logs.logMessage(this.logs.INFO, sys.name(src) + " logged in.");
 68 
 69          return;
 70      },
 71 
 72 
 73      beforeLogOut: function (src)
 74      {
 75          this.logs.logMessage(this.logs.INFO, sys.name(src) + " logged out.");
 76      },
 77 
 78      afterChangeTeam: function (src)
 79      {
 80         // this.profile.registerPlayer(src);
 81      }
 82  });
