233     equipQMult: function (e)
234     {
235         if (! ("quality" in e)) return 0; // obvious error but wont cause crash
236 
237         if (e.quality === null) return 10; // items blessed by tux (hacked dev toys >:D)
238 
239         else return 1 - (1 / Math.log(e.quality/100 + Math.E));
240     }
241     ,
242     equipAtk: function (e)
243     {
244         if (!e) return 0;
245         return Math.max(20, (this.equips[e.type].base || 20) * (this.materials[e.material].sharpness || 20) * this.equipQMult(e));
246     }
247     ,
248     equipDef: function (e)
249     {
250         if (!e) return 0;
251         return Math.max(20, (this.equips[e.type].base || 20) * (this.materials[e.material].strength || 20) * this.equipQMult(e));
252     }
253     ,
254     equipName: function(e)
255     {
256         if (!e) return "Nothing";
257         var qm = this.equipQMult(e);
258         var qs;
259         var qdt = [
260             [0, "Failure of a %"],
261             [0.3, "Really shity %"],
262             [0.5, "Ordinary %"],
263             [0.70, "Fine %"],
264             [0.82, "Excellent %"],
265             [0.87, "Superb %"],
266             [0.92, "Supreme %"],
267             [0.94, "Legendary %"],
268             [0.97, "Sacred %"],
269             [1, "Divine % (hacked >:3)"]
270         ];
271 
272         for (var x in qdt) if (qm >= qdt[x][0]) qs = qdt[x][1];
273 
274 
275         return qs.replace(/\%/, (this.materials[e.material].name + " " + this.equips[e.type].name));
276     }
