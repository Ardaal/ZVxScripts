 44      fetch: function (modname, source)
 45      {
 46          if (source.type === "local")
 47          {
 48              if (source.modules && source.modules.indexOf(modname) === -1) return null;
 49 
 50              try
 51              {
 52                  var text = sys.read(source.path + "/" + modname + ".js");
 53 
 54                  if (!source.sha1 || source.sha1[modname] === sys.sha1(text)) return text;
 55 
 56 
 57                  try
 58                  {
 59                      this.modules.logs.logMessage(this.modules.logs.WARN, "Invalid sha1 string for module.");
 60                  }
 61                  catch (_)
 62                  {
 63                      print("Invalid sha1 string for module.");
 64                  }
 65                  return null;
 66 
 67              }
 68              catch (e)
 69              {
 70                  return null;
 71              }
 72          }
 73 
 74          if (source.type === "web")
 75          {
 76              if (source.modules && source.modules.indexOf(modname) === -1) return null;
 77 
 78              var text = sys.synchronousWebCall(source.path + "/"+modname + ".js");
 79 
 80              if (!text) return null;
 81 
 82              if (!source.sha1 || source.sha1[modname] === sys.sha1(text)) return text;
 83 
 84              try
 85              {
 86                  this.modules.logs.logMessage(this.modules.logs.WARN, "Invalid SHA1 string for module.");
 87              }
 88              catch (_)
 89              {
 90                  print("Invalid SHA1 string for module.");
 91              }
 92              return null;
 93 
 94          }
 95 
 96          else return null;
 97      },

130      loadModule: function (modname)
131      {
132          var modinf = this.moduleTable[modname];
133 
134          if (!modinf) modinf = this.moduleTable[modname] = this.statModule(modname);
135 
136          if (!modinf) return; // failure
137 
138          if (modinf.state === "loaded" || modinf.state === "error") return;
139 
140          if (modinf.state === "ready" || modinf.state === "pending")
141          {
142              this.enableModule(modname, modinf.code);
143              return;
144          }
145 
146          if (modinf.state === "unloaded")
147          {
148 
149              var text = modinf.text || this.fetch(modname, modinf.source);
150 
151              if (!text) return;
152 
153              var code = null;
154              var extratext = null;
155              var extra;
156 
157              try
158              {
159                  try { code = sys.eval(text, modinf.source.path +"/"+ modname + ".js"); }
160                  catch(_) { code = sys.eval(text); }
161 
162                  if (code.include) for (var x in code.include)
163                  {
164                      var i = code.include[x];
165 
166                      extratext = this.fetch(code.include[x], modinf.source);
167 
168                      if (!extratext) return;
169 
170                      try { extra = sys.eval(extratext, modinf.source.path +"/"+ modname + ".js"); }
171                      catch(_) { extra = sys.eval(extratext); }
172 
173                      this.fuse(code, extra);
174                  }
175 
176                  extra = null; extratext = null;
177 
178                  if (code.initializeModule) code.initializeModule();
179 
180                  modinf.code = code;
181                  modinf.pendingCode = null;
182 
183                  if (this.enableModule(modname, modinf.code))
184                  {
185                      modinf.state = "loaded";
186 
187                      if (this.log) this.log("Loading module " + modname);
188                  }
189                  else
190                  {
191 
192                      modinf.state = "error";
193                      print("error");
194                  }
195 
196 
197              }
198              catch(e)
199              {
200                  modinf.state = "error";
201                  print(e);
202              }
203 
204          }
205      },

253      /** Enables a module: Takes the module object and glues it into to the system.
254       * @param modname The name of the module
255       * @param module The module object
256       */
257      enableModule: function (modname, module)
258      {
259          var x;
260 
261          module.submodules = [];
262          Object.defineProperty(module, "script", {value: this, configurable: true});
263 
264          this.modules[modname] = module;
265 
266          if (!module.require) module.require = [];
267 
268          for (x in module.require)
269          {
270              var i = module.require[x];
271 
272              delete module[i]; // clean up old refs
273 
274              this.loadModule(i);
275 
276              if (!this.moduleTable[i] || this.moduleTable[i].state !== "loaded" || !this.modules[i])
277              {
278                  this.moduleTable[modname].state = "pending";
279 
280                  delete this.modules[modname];
281 
282                  return false;
283 
284              }
285 
286 
287              if (this.modules[i].submodules.indexOf(modname) !== -1) throw new Error("Corrupt module system.");
288 
289              // Add this module to submodules of the other module
290              this.modules[i].submodules.push(modname);
291 
292              // Apply requirements:
293              Object.defineProperty(this.modules[modname], i, {configurable: true, value: this.modules[i]});
294 
295          }
296 
297 
298          for (x in this.hooks)
299          {
300              module[x] = this.hooks[x];
301          }
302 
303 
304          for (x in module.require)
305          {
306              var i = module.require[x];
307 
308              if (this.modules[i].loadSubmodule) this.modules[i].loadSubmodule(modname, module);
309          }
310 
311 
312          if (module.loadModule) module.loadModule();
313 
314          return true;
315 
316      },

322          if (! (modname in this.modules)) return;
323 
324          var disabled = [];
325 
326          var module = this.modules[modname];
327 
328          var submodules = [];

334          if (module.unloadModule) module.unloadModule();
335 
336          if (module.unloadModuleHooks)
337          {
338              var unloadModuleHooks = module.unloadModuleHooks;
339 
340              for (x in unloadModuleHooks)
341              {
342                  unloadModuleHooks[x].apply(new Object, [module]);
343              }
344 
345          }
346 
347 
348          delete this.modules[modname];
349 
350          this.moduleTable[modname].state = "ready";
351 
352          if (this.log) this.log("Unloaded module: " + modname);
353

356      fuse: function (origin, extra)
357      {
358          for (var x2 in extra)
359          {
360              if (x2 in origin && origin[x2] != null)
361              {
362                  if (typeof origin[x2] != typeof extra[x2] || (typeof origin[x2] != "object" && typeof origin[x2] != "function"))
363                  {
364                      throw new Error("Unable to merge");
365                  }
366                  if (typeof origin[x2] === "function")
367                  {
368                      // use a closure to merge the two functions as one
369                      origin[x2] = (function (a, b) {
370                                     return function ()
371                                     {
372                                         a.apply(origin, arguments);
373                                         b.apply(origin, arguments);
374                                     };
375                                 })(origin[x2], extra[x2]);
376                  }
377                  else if (origin[x2] instanceof Array)
378                  {
379                      origin[x2] = origin[x2].concat( extra[x2] );
380                  }
381                  else
382                  {
383                      for (var x3 in extra[x2])
384                      {
385                          if (x3 in origin[x2]) throw new Error("Unable to merge");
386 
387                          origin[x2][x3] = extra[x2][x3];
388                      }
389                  }
390              }
391              else
392              {
393                  origin[x2] = extra[x2];
394              }
395          }
396      },
