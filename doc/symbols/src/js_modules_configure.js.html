 32      {
 33          server: true,
 34 
 35          code: function (src, cmd, chan, cache)
 36          {
 37              try
 38              {
 39                  var groups = cache || this.user.groups(src);
 40                  if (this.parseConfigureString(src, cmd.input || "", groups) != 2) this.com.message(src, "Configured.", this.theme.INFO);
 41                  return;
 42              }
 43              catch (e)
 44              {
 45                  this.com.message(src, "Configure error in " + cmd.input + ": " + e, this.theme.WARN);
 46 
 47 
 48 
 49                  return;
 50              }
 51          },
 52 
 53          perm: function (src)
 54          {
 55              return this.user.hasPerm(src, "CONFIGURE");
 56 
 57          }
 58 
 59 
 60      },
 61 
 62      parseConfigureString: function (src, string)
 63      {
 64          var match = string.match(/^\s*([\w]+(?:\.[\w]+)*)(?:\s*(\=|\+\=|<<|>>|~)\s*(.*$))?/i);
 65 
 66          //if (this.user.isSuper(src)
 67          //implement permission checks later
 68 
 69          if (!match)
 70          {
 71              this.com.message(src, "Server Configuration:\n" + JSON.stringify(this.io.configs, 1, " "), this.theme.INFO);
 72              return 2;
 73          }
 74 
 75          var pathway = match[1].split(/\./g);
 76 
 77          if (pathway.length <= 1 && (match[2] || pathway.length == 0)) throw new Error("Inconfigurable property");
 78 
 79 
 80 	 var modname = pathway[0];
 81          var finalProp = pathway.pop();
 82 
 83          if (finalProp === "__proto__") throw new Error("Smartass eh?");
 84 
 85          var modobj = this.io.configs;
 86          // var schemaobj = this.io.schemas[module];
 87          // var schema;
 88 
 89          for (var x in pathway)
 90          {
 91 
 92              var i = pathway[x];
 93 
 94              if (i === "__proto__") throw new Error("Smartass eh?");
 95 
 96              modobj = modobj[pathway[x]];
 97 
 98              if (typeof modobj !== "object" || modobj instanceof Array) throw new Error("Invalid property chain");
 99 
100          }
101 
102          if (! match[2])
103          {
104              this.com.message(src, "Property Value:\n" + JSON.stringify(modobj[finalProp], 1, " "), this.theme.INFO);
105              return 2;
106          }
107 
108          if (!this.user.hasPerm(src, "CONFIGURE[" + match[1].toUpperCase() + "]"))
109          {
110              throw new Error("Permission denied.");
111          }
112 
113          var prop = JSON.parse(match[3]);
114 
115 
116 
117          switch (match[2])
118          {
119          case ":=":
120          case "=":
121              if (typeof prop != "object" && typeof prop === typeof modobj[finalProp])
122              {
123                  modobj[finalProp] = prop;
124                  this.io.callConfigureHooks(modname);
125                  return true;
126              }
127              else if (typeof modobj[finalProp] === "object" && modobj[finalProp] instanceof Array && typeof prop === "object" && prop instanceof Array)
128              {
129                  for (x in prop) if (typeof prop[x] === "object") throw new Error("Wrong argument type or wrong operator.");
130 
131                  modobj[finalProp] = prop;
132                  this.io.callConfigureHooks(modname);
133                  return true;
134              }
135              throw new Error("Wrong argument type or wrong operator.");
136          case "<<":
137              if (typeof modobj[finalProp] === "object" && modobj[finalProp] instanceof Array && typeof prop != "object")
138              {
139                  modobj[finalProp].push(prop);
140                  this.io.callConfigureHooks(modname);
141                  return true;
142              }
143              throw new Error("Wrong argument type or wrong operator.");
144          case ">>":
145              if (typeof modobj[finalProp] === "object" && modobj[finalProp] instanceof Array && typeof prop != "object")
146              {
147                  if (modobj[finalProp].indexOf(prop) !== -1) modobj[finalProp].splice(modobj[finalProp].indexOf(prop), 1);
148                  this.io.callConfigureHooks(modname);
149                  return true;
150              }
151              throw new Error("Wrong argument type or wrong operator.");
152          case "~":
153 
154 
155 
156 
157          }
158 
159          throw new Error("Wrong argument type or wrong operator.");
160
