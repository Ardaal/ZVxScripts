 22 /** An rpg battle descriptor. Note the format is subject to change.
 23  * @name rpgBattle
 24  * @class
 25  */
 26 /** The players in the battle. Note this is an array of player /names/ not objects. This is done so as not to have duiplicates of player objects in the database.
 27  * @name players
 28  * @type {Array.<String>}
 29  * @memberOf rpgBattle.prototype
 30  */
 31 /** The mobs in the battle.
 32  * @name mobs
 33  * @type {Array.<rpgMob>}
 34  * @memberOf rpgBattle.prototype
 35  */
 36 /** @scope script.modules.rpg_game */
 37 ({
 38     /** Battle step will cause a turn in a battle to take place
 39      * @event
 40      * @param {rpgCtx} ctx
 41      * @param {rpgBattle} ctx.battleid The id of the specific battle this should step.
 42      * @param {rpgClass} ctx.rpg The current rpg we are in.
 43      */
 44     battleStep: function (ctx)
 45     {
 46         var x;
 47         var rpg = ctx.rpg;
 48 
 49         var battle = ctx.battle;
 50 
 51         // team_players is an array of all the players that are playing
 52         // we make it from battle.players
 53         var team_players = [];
 54         for (var x in battle.players)
 55         {
 56             team_players.push(rpg.players[battle.players[x]]);
 57             // Battle players contains NAMES of players, not the player objects!
 58         }
 59 
 60         var team_mobs = []; // Do not save!
 61         for ( x in battle.mobs)
 62         {
 63             team_mobs.push(battle.mobs[x]);
 64         }
 65 
 66         var entities = []; // Do not save!
 67 
 68 
 69         for (x in team_players)
 70         {
 71             entities.push({type: "player", e: team_players[x]});
 72         }
 73 
 74         for (x in team_mobs)
 75         {
 76             entities.push({type: "mob", e: team_mobs[x]});
 77         }
 78 
 79         entities.sort(
 80             function _sorting_function_ (a, b)
 81             {
 82                 return a.speed - b.speed;
 83             }
 84         );
 85 
 86         var pids = [];
 87 
 88         for ( x in battle.players) pids.push(sys.id(battle.players[x]));
 89 
 90         battleLoop: for ( x in entities)
 91         {
 92             var attacker = ctx.attacker = entities[x].e;
 93             //ctx.attackerIsPlayer = entities[x].type == "player";
 94             var move = ctx.move = this.pickMove(entities[x]);
 95 
 96             if (ctx.move.cost) for (var x2 in ctx.move.cost)
 97             {
 98                 ctx.attacker[(x2 === "mp"? "mana" : x)] -= ctx.move.cost[x2];
 99                 if (ctx.attacker[(x2 === "mp"? "mana" : x)] < 0)
100                 {
101                     this.com.message(pids, ctx.attacker.name + " tried to use "  + ctx.move.name + " but didn't have enough " + this.longStatName[x2 === "mp" ? "mana" : x2],
102                                     this.theme.GAME);
103                     continue battleLoop;
104                 }
105             }
106 
107             for (var x2 in ctx.move.components)
108             {
109                 var cmp = ctx.move.components[x2];
110                 var targets = ctx.targets = [];
111                 var count = cmp.count;
112                 switch (typeof cmp.target)
113                 {
114                 case "object":
115                     // Array
116                     if (cmp.target.indexOf("ally") != -1) ctx.targets = ctx.targets.concat(team_players);
117                     if (cmp.target.indexOf("opp") != -1) ctx.targets = ctx.targets.concat(team_mobs);
118                     if (cmp.target.indexOf("self") != -1 && ctx.indexOf(ctx.attacker) == -1) ctx.targets.push(ctx.attacker);
119                     break;
120                 case "string":
121                     switch (ctx.move.type)
122                     {
123                         case "ally": ctx.targets = team_players;
124                         case "self": ctx.targets = [attacker];
125                     }
126                 }
127 
128                 this.util.shuffle(targets);
129 
130                 if (count > 0) for (var x3 in targets)
131                 {
132                     if (count-- === 0) break;
133 
134 
135                     this.moves[cmp.move]({attacker: entities[x], targets:targets, component:cmp});
136 
137 
138 
139                 }
140             }
141         }
142 
143     }
144 
145 });
