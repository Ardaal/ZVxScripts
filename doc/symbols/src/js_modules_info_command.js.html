 22 ({
 23     require: ["com", "text", "commands", "theme", "logs", "user"]
 24 
 25     ,
 26 
 27     info:
 28     {
 29         server: true
 30         ,
 31         desc: "Shows information about users"
 32         ,
 33         perm: "INFO"
 34         ,
 35         code: function (src, cmd)
 36         {
 37             var names = cmd.args;
 38 
 39 
 40             var se_group_perm = this.user.hasPerm(src, "INFO[PERMS]");
 41             var mj_group_perm = this.user.hasPerm(src, "INFO[GROUPS]");
 42             var auth_perm = this.user.hasPerm(src, "INFO[AUTH]");
 43             var ip_perm = this.user.hasPerm(src, "INFO[IP]");
 44 
 45             var m = [];
 46 
 47             for (var x in names)
 48             {
 49                 m.push("<b>Info about user: " + names[x] +"</b>");
 50                 if (auth_perm) m.push("Auth level: " + this.user.nameAuth(names[x]));
 51                 if (ip_perm) m.push("IP Address: " + sys.dbIp(names[x]));
 52                 if (mj_group_perm) m.push("Major Groups: " + Object.keys(this.user.nameMajorGroups(names[x])).join (", ") + ".");
 53                 if (se_group_perm) m.push("Permissions: " + Object.keys(this.user.nameGroups(names[x])));
 54                 //m.push("Registered: " + sys.dbRegistered(names[x]));
 55             }
 56 
 57             this.com.message([src], "<br/>"+m.join("<br/>"), this.theme.INFO, true);
 58         }
 59     }
 60     ,
 61     loadModule: function()
 62     {
 63         this.commands.registerCommand("info", this);
 64     }
 65 });
