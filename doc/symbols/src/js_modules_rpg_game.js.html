 22 /** Contains the RPG game :)
 23  * @name rpg_game
 24  * @memberOf script.modules
 25  * @requires io, com, theme, commands, util, logs, less
 26  * @namespace
 27  * */
 28 (
 29 /** @scope script.modules.rpg_game */
 30 {
 31     /** List of modules to be required.
 32      * @type {string[]}
 33      */
 34     require: ["io", "com", "theme", "commands", "util", "logs", "less"]
 35     ,
 36     /** List of files to be included in this module.
 37      * @type {string[]}
 38      */
 39     include: ["rpg_areas", "rpg_player", "rpg_entity", "rpg_actions",
 40               "rpg_mobs", "rpg_equips", "rpg_materials", "rpg_moves",
 41              "rpg_battle"]
 42     ,
 43     /** I/O database that keeps track of games etc.
 44      * @type {IOdatabase}
 45      */
 46     database: null
 47     // database stores the permanent data, games, etc.
 48     ,
 49     /** channel */
 50     channels: null
 51     // channels is used to associate databases with channels, it doesn't keep between reloads
 52     ,
 53     hooks: null
 54     // hooks associates rpgs with pipes
 55     // not used as it should be for right now, will be fixed later
 56     ,
 57     loadModule: function ()
 58     {
 59         this.database = this.io.openDB("rpg_game");
 60         if (!this.database.games) this.database.games = new Object;
 61         this.channels = new Object;
 62         this.hooks = new Object;
 63 
 64         this.commands.registerCommand("loadrpg", this);
 65         this.commands.registerCommand("rpg", this);
 66 
 67         this.script.registerHandler("step", this);
 68     }
 69     ,
 70     unloadModule: function ()
 71     {
 72         this.io.closeDB("rpg_game");
 73     }
 74     ,
 75     step: function ()
 76     {
 77         for (var x in this.channels)
 78         {
 79             this.RPGStep(this.channels[x]);
 80         }
 81     }
 82     ,
 83     RPGStep: function (rpg)
 84     {
 85         if (rpg.paused) return;
 86         rpg.tick++;
 87 
 88         for (var x in rpg.areas)
 89         {
 90             this.areaStep(rpg.areas[x], { rpg: rpg });
 91         }
 92 
 93         for (x in rpg.players)
 94         {
 95             this.playerStep(rpg.players[x], {rpg: rpg});
 96         }
 97 
 98         for (x in rpg.battles)
 99         {
100             this.rpgStep( {rpg:rpg, battle: rpg.battles[x] });
101         }
102     }
103     ,
104     RPG: function (rpgname)
105     {
106         if (this.database.games[rpgname]) { return this.database.games[rpgname]; }
107 
108         else
109         {
110             var newr = this.database.games[rpgname] =
111                 {
112                     name: rpgname,
113                     areas: JSON.parse(JSON.stringify(this.areas)),
114                     materials: {},
115                     players: {},
116                     battles: {},
117                     battleCoutner: 0,
118                     running: false,
119                     tick: 0
120                 };
121 
122             for (var x in newr.areas)
123             {
124                 newr.areas[x].battles = [];
125             }
126 
127 	    return newr;
128         }
129 
130         return newr;
131     }
132     ,
133     startRPGinChan: function (rpgname, chan)
134     {
135         var rpg = this.database.games[rpgname];
136 
137         this.channels[chan] = rpg;
138 
139         if ( this.hooks[rpgname] ) throw new Error("Already hooked into another channel?");
140 
141         this.logs.logMessage(this.logs.INFO, "Loaded RPG game " + rpgname + " in channel " + sys.channel(chan));
142 
143         this.hooks[rpgname] =
144             {
145                 message: this.util.bind
146                 (
147                     this
148                     ,
149                     function(player, message)
150                     {
151                         this.com.message([player], message, this.theme.GAME, true, [chan]);
152                     }
153                 )
154                 ,
155                 broadcast: this.util.bind
156                 (
157                     this
158                     ,
159                     function (message)
160                     {
161                         this.com.broadcast(message, this.theme.GAME, true, [chan]);
162                     }
163                 )
164             };
165 
166 
167         this.hooks[rpgname].running = true;
168 
169     }
170     ,
171     /** @type commandDescriptor */
172     rpg: // not to be confused with "RPG"
173     {
174         desc: "does rpg stuffs",
175         perm: function ()
176         {
177             return true;
178         }
179         ,
180         code: function (src, cmd, chan)
181         {
182             if (!this.channels[chan])
183             {
184                 this.com.message([src], "No RPG running in that channel", this.theme.WARN);
185                 return;
186             }
187 
188             var rpg = this.channels[chan];
189 
190             var player = rpg.players[sys.name(src).toLowerCase()];
191             if (! player)
192             {
193                 this.com.message([src], "Creating you a new RPG character!", this.theme.GAME);
194                 this.logs.logMessage(this.logs.INFO, sys.name(src) + " created an RPG character in RPG " + rpg.name);
195 
196                 player = this.newPlayer();
197                 player.name = sys.name(src).toLowerCase();
198 
199                 rpg.players[sys.name(src).toLowerCase()] = player;
200             }
201 
202             var actions = String(cmd.input).split(/\;/g);
203 
204             for (var x in actions)
205             {
206                 var subactions = actions[x].split(/[,:]/g);
207 
208                 if (subactions.length == 0) continue;
209 
210                 subactions[0] = subactions[0].toLowerCase();
211                 if (subactions[0] in this.rpgActions)
212                 {
213                     this.rpgActions[subactions[0]].apply(this, [src, subactions, chan, {player:player, rpg: rpg }] );
214                 }
215             }
216         }
217     }
218     ,
219     /** @type commandDescriptor */
220     loadrpg:
221     {
222         desc: "Loads an RPG into a channel."
223         ,
224         perm: function (src)
225         {
226             return sys.auth(src) >= 2;
227         }
228         ,
229         code: function (src, cmd, chan)
230         {
231 
232             var rpgname = cmd.input.replace(/^\s*([^\s]+)\s*$/, "$1");
233 
234             if (this.channels[chan])
235             {
236                 this.com.message([src], "This channel already has a running RPG", this.theme.WARN);
237                 return;
238             }
239 
240             var rpg = this.RPG(rpgname);
241             if (this.hooks[rpgname])
242             {
243                 this.com.message([src], "RPG is already running!", this.theme.WARN);
244                 return;
245             }
246 
247             this.startRPGinChan(rpgname, chan);
248         }
249     }
250 
251 });
