1 ({ // starts module
  2     require: ["commands", "com", "io"],
  3     loadModule: // rules.loadModule
  4     function ()
  5     {
  6         this.commands.registerCommand("motd", this);
  7         this.io.registerConfig(this, { message: "", sendMOTDOnLogin: true });
  8         this.script.registerHandler("afterLogIn", this);
  9     },
 10     motd:
 11     {
 12         perm: function ()
 13         {
 14             return true; // all users shoudl be able to use /motd so don't check anything
 15         },
 16         code: function (src, cmd, chan) // code that runs /motd
 17         {
 18              this.com.message(src, "MOTD: " + this.config.message);
 19         }
 20     },
 21     afterLogIn: function (src)
 22     {
 23         if (this.config.sendMOTDOnLogin )
 24         {
 25             this.com.message(src,"MOTD: " + this.config.message);
 26         }
 27     }
 28 });
