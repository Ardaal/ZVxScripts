 22 ({
 23     require: ["sched", "io", "profile", "com"]
 24     ,
 25     database: null
 26     ,
 27     loadModule: function () 
 28     {
 29         this.database = this.io.openDB("security");
 30 
 31         if (!this.database.bans) this.database.bans = new Object;
 32 
 33         if (!this.database.mutes) this.database.mutes = new Object;
 34 
 35         var _this = this;
 36 
 37         for (var x in this.database.bans)
 38         {
 39             var b = this.profile.trace(x);
 40 
 41             // todo merge bans properly
 42             if (b != x)
 43             {
 44                 this.database.bans[b] = this.database.bans[b] || this.database.bans[x];
 45             }
 46             var ban = this.database.bans[x];
 47          
 48             if (ban.expires)
 49             {
 50                 (function(x, _this) {
 51                     _this.sched.at(ban.expires, function () { _this.checkBan(x); });
 52                 })(x, this);
 53             }
 54         }
 55 
 56         for (var x in this.database.mutes)
 57         {
 58             var mute = this.database.mutes[x];
 59             var b = this.profile.trace(x);
 60             if (b != x)
 61             {
 62                 this.database.mutes[b] = this.database.mutes[b] || this.database.mutes[x];
 63             }
 64          
 65             if (mute.expires)
 66             {
 67                 (function(x, _this) {
 68                     _this.sched.at(mute.expires, function () { _this.checkMute(x); });
 69                 })(x, this);
 70             }
 71         }
 72     }
 73     ,
 74     unloadModule: function()
 75     {
 76         this.io.closeDB("security");
 77     }
 78     ,
 79     checkUsers: function (src)
 80     {
 81         var uids = sys.playerIds();
 82 
 83         for (var x in uids)
 84         {
 85             this.checkUser(uids[x]);
 86         }
 87     }
 88     ,
 89     checkUser: function (src)
 90     {
 91         var p = this.profile.profileID(src);
 92 
 93         if (sys.auth(src) != 3 && this.profIsBanned(p))
 94         {
 95             sys.kick(src);
 96         }
 97     }
 98     ,
 99     profIsMuted: function (p)
100     {
101         if (p in this.database.mutes) return true;
102 
103         return false;
104     } 
105     ,
106     getMute: function (profid)
107     {
108         return this.database.mutes[profid];
109     }
110     ,
111     setMute: function (profid, mute)
112     {
113         this.database.mutes[profid] = mute;
114 
115         var _this = this;
116         if (mute.expires)
117         {
118             this.sched.at(mute.expires, function () { _this.checkMute(profid); });
119         }
120 
121         this.io.markDB("security");
122     }
123     ,
124 
125     checkMute: function (profid)
126     {
127         var mute = this.database.mutes[profid];
128 
129         if (!mute) return;
130 
131         if (mute.expires && mute.expires <= +new Date)
132         {
133             this.com.broadcast("Mute on " + this.profile.lastName(profid) + " (#: "+profid+") expired.");
134             delete this.database.mutes[profid];
135         }
136     }
137     ,
138     removeMute: function (profid)
139     {
140         delete this.database.mutes[profid];
141 
142         this.io.markDB("security");
143     }
144     ,
145     profIsBanned: function (p)
146     {
147         if (p in this.database.bans) return true;
148 
149         return false;
150     } 
151     ,
152     setBan: function (profid, ban)
153     {
154         this.database.bans[profid] = ban;
155         var _this = this;
156         if (ban.expires)
157         {
158             this.sched.at(ban.expires, function () { _this.checkBan(profid); });
159         }
160 
161         this.io.markDB("security");
162     }  
163     ,
164     removeBan: function(profid)
165     {
166         delete this.database.bans[profid];
167 
168         this.io.markDB("security");
169     }
170     ,
171     getBan: function (profid)
172     {
173         return this.database.bans[profid];
174     }  
175     ,
176     checkBan: function (profid)
177     {
178         var ban = this.database.bans[profid];
179 
180         if (!ban) return;
181 
182         if (ban.expires && ban.expires <= +new Date)
183         {
184             this.com.broadcast("Ban on " + this.profile.lastName(profid) + " (#: "+profid+") expired.");
185             delete this.database.bans[profid];
186         }
187     }
188 })
