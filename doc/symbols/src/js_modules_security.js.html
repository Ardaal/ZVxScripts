 22 ({
 23      require: ["sched", "io", "com", "user", "theme", "util", "text"],
 24 
 25      database: null,
 26 
 27      loadModule: function ()
 28      {
 29          this.database = this.io.registerDB(this, "security.2");
 30 
 31          this.script.registerHandler("beforeLogIn", this);
 32          this.script.registerHandler("afterChangeTeam", this);
 33 
 34          this.database.bans = this.database.bans || new Object;
 35          this.database.mutes = this.database.mutes || new Object;
 36 
 37          this.database.muteCtr = this.database.muteCtr || 0;
 38 
 39          this.database.banCtr = this.database.banCtr || 0;
 40 
 41      },
 42 
 43      chk: function ()
 44      {
 45          var q = this.user.users();
 46 
 47          if (this.script.modules.gateway) for (var x in q)
 48          {
 49              this.script.modules.gateway.checkUser(q[x]);
 50          }
 51      },
 52 
 53 
 54      afflicted: function (src, object)
 55      {
 56          var lname, ip;
 57 
 58          ip = (src == 0? "0.0.0.0" : sys.ip(src));
 59 
 60          lname = this.user.name(src).toLowerCase();
 61 
 62         // if (this.user.hasPerm(src, "PROTECTED")) return null;
 63 
 64          for (var x in object)
 65          {
 66              var ban = object[x];
 67 
 68              if (ban.names) for (var x2 in ban.names)
 69              {
 70                  if (lname == ban.names[x2].toLowerCase())
 71                  {
 72                      return x;
 73                  }
 74 
 75              }
 76 
 77              if (ban.nameRegex) for (var x2 in ban.nameRegex)
 78 
 79 
 80              {
 81                  var m = ban.nameRegex[x2].match(/^\/(.+)\/(\w+)$/);
 82                  if (lname.match(new RegExp(m[1], m[2]))) return x;
 83 
 84              }
 85 
 86              if (ban.ips) for (var x2 in ban.ips)
 87              {
 88                  if (ip == ban.ips[x2]) return x;
 89              }
 90 
 91              if (ban.subnets) for (var x2 in ban.subnets)
 92              {
 93                  if (this.util.ipMatchesSubnet(ip, ban.subnets[x2]))
 94                  {
 95                      return x;
 96 
 97                  }
 98 
 99              }
100          }
101 
102          return null;
103      },
104 
105      removeAfflicted: function (param, object)
106      {
107          var lname, ip;
108 
109          var rs = [];
110 
111          lname = param.toLowerCase();
112 
113 
114          loop1: for (var x in object)
115          {
116              var ban = object[x];
117 
118              if (ban.names) for (var x2 in ban.names)
119              {
120                  if (lname == ban.names[x2].toLowerCase())
121                  {
122                      rs.push(x);
123                      delete object[x];
124                      continue loop1;
125                  }
126 
127              }
128 
129              if (ban.nameRegex) for (var x2 in ban.nameRegex)
130 
131 
132              {
133                  var m = ban.nameRegex[x2].match(/^\/(.+)\/(\w+)$/);
134                  if (lname.match(new RegExp(m[1], m[2])) || param == ban.nameRegex[x2])
135                  {
136                      rs.push(x);
137                      delete object[x];
138                      continue loop1;
139                  }
140 
141              }
142 
143              if (ban.ips) for (var x2 in ban.ips)
144              {
145                  if (param == ban.ips[x2])
146                  {
147                      rs.push(x);
148                      delete object[x];
149                      continue loop1;
150                  }
151              }
152 
153              if (ban.subnets) for (var x2 in ban.subnets)
154              {
155                  if (param == ban.subnets[x2] || (param.match(/\d+\.\d+\.\d+\.\d+/) && this.util.ipMatchesSubnet(param, ban.subnets[x2])))
156                  {
157                      rs.push(x);
158                      delete object[x];
159                      continue loop1;
160                  }
161 
162              }
163          }
164 
165          return rs;
166      },
167 
168      banID: function (src)
169      {
170          return this.afflicted(src, this.database.bans);
171      },
172 
173      muteID: function (src)
174      {
175          return this.afflicted (src, this.database.mutes);
176      },
177 
178      beforeLogIn: function (src)
179      {
180          var name = this.user.name(src);
181 
182          if (name.match(/\d+\.\d+.\d+\.\d+/) || name.match(/\//))
183          {
184              this.com.message(src, "Illegal name.", this.theme.CRITICAL);
185              sys.stopEvent();
186              return;
187          }
188      },
189 
190      afterChangeTeam: function (src)
191      {
192 
193          var name = this.user.name(src);
194 
195          if (name.match(/\d+\.\d+.\d+\.\d+/) || name.match(/\//))
196          {
197              this.com.message(src, "Illegal name.", this.theme.CRITICAL);
198              sys.kick(src);
199              return;
200          }
201      }
