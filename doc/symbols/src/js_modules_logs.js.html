 22 ({
 23     require: ["com", "theme", "text", "util"]
 24     ,
 25     logs: []
 26     ,
 27     DEBUG: 0, USER:1, INFO: 2, WARN: 3, ERROR: 4, CRITICAL: 5
 28     ,
 29     logMessage: function (level,msg)
 30     {
 31         
 32         if (level != 0) print("logmsg lv" + level + ": " +msg);
 33         this.logs.push([level, msg]);
 34 
 35         if (level >= 3){
 36             var auths = [];
 37             sys.playerIds().forEach(function(i) { if (sys.auth(i) >= 1) auths.push(i); });
 38 
 39             this.com.message(auths, msg, this.theme.CRITICAL);
 40         } 
 41         else if (level >= 2)
 42         {
 43             var auths = [];
 44             sys.playerIds().forEach(function(i) { if (sys.auth(i) >= 1) auths.push(i); });
 45 
 46             this.com.message(auths, msg, this.theme.INFO);
 47         } 
 48         else if (1 || level == 1)
 49         {
 50             var auths = [];
 51             sys.playerIds().forEach(function(i) { if (sys.auth(i) == 3) auths.push(i); });
 52 
 53             this.com.message(auths, msg, this.theme.INFO);   
 54         }
 55         
 56         try 
 57         {
 58             sys.append("logs.txt", JSON.stringify([level, msg, (new Date).toString()]) + "\n");
 59         } catch (_) {}
 60     }
 61     ,
 62     loadModule: function ()
 63     {
 64         this.savedLogFunction = script.log;
 65         script.log = this.util.bind(
 66             this
 67             ,
 68             function(msg)
 69             {
 70                 this.logMessage(this.INFO, msg);
 71             }
 72         );
 73     }
 74     ,
 75     unloadModule: function ()
 76     {
 77         script.log = this.savedLogFunction;
 78     }
 79 });
