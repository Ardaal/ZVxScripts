 90           * @param {string} handlername The event name of the script handler.

 92           * @param {string} [propname=handlername] The name of the handler on the module, if it is different from the event name.

 94          registerHandler: function registerHandler (handlername, object, propname)
 95          {
 96              if (!propname) propname = handlername;
 97 
 98              if (! (handlername in script))
 99              {
100                  var f = function _meta_event_handler_func_ ()
101                  {
102                      try
103                      {
104                          for (var x in f.callbacks)
105                          {
106                              f.callbacks[x].func.apply(f.callbacks[x].bind, arguments);
107                          }
108                      }
109 
110                      catch (err)
111                      {
112                          this.error(err);
113                      }
114                  };
115 
116                  script[handlername] = f;
117                  script[handlername].callbacks = [];
118              }
119 
120              if ( !(script[handlername].callbacks)) throw new Error("Not registerable");
121 
122              var callbk = {func:object[propname], bind:object};
123 
124              script[handlername].callbacks.push(callbk);
125 
126              var _bind = this;
127              if ("onUnloadModule" in object)
128              {
129                  object.onUnloadModule(
130                      function _meta_callback_unload_ ()
131                      {
132                          script[handlername].callbacks.splice(script[handlername].callbacks.indexOf(callbk),1);
133                      }
134                  );
135              }
136 
137              return;

142          {
143              var oldmod = this.modules[modname];
144 
145              if (!oldmod) return false;
146 
147              if (!oldmod.hotswap) return false;
148 
149              var newMod = sys.exec("js_modules/" + modname + ".js");
150 
151              if (!newMod.hotswap) return false;
152 
153              this.log("Hotswapping module: " + modname);
154 
155              Object.defineProperty(newMod, "script", {value: this, configurable: true});
156 
157              for (var x in newMod.require)
158              {
159                  this.loadModule(newMod.require[x]);
160                  Object.defineProperty(newMod, newMod.require[x], {value: this.modules[newMod.require[x]], configurable: true});
161                  if (this.modules[newMod.require[x]].submodules.indexOf(modname) == -1) this.modules[newMod.require[x]].submodules.push(modname);
162              }
163 
164              function nil(){};
165 
166              switch (typeof newMod.hotswap)
167              {
168              case "function":
169                  if (newMod.hotswap(oldmod))
170                  {
171                      if (oldmod.submodules) for (var x in oldmod.submodules)
172                      {
173                          delete this.modules[oldmod.submodules[x]][modname];
174                          Object.defineProperty(this.modules[oldmod.submodules[x]], modname, {value: newMod, configurable: true});
175                      }
176 
177                      newMod.submodules = this.modules[modname].submodules;
178 
179                      this.modules[modname] = newMod;
180 
181 
182                      return true;
183                  }
184                  else return false;
185              case "boolean":
186                  (oldmod.unloadModule || nil) ();
187 
188                  if (oldmod.submodules) for (var x in oldmod.submodules)
189                  {
190                      delete this.modules[oldmod.submodules[x]][modname];
191                      Object.defineProperty(this.modules[oldmod.submodules[x]], modname, {value: newMod, configurable: true});
192                  }
193 
194                  newMod.submodules = oldmod.submodules;
195                  this.modules[modname] = newMod;
196                  (newMod.loadModule || nil)();
197                  return true;
198              default:
199                  return false;
200              }

221          /** Loads a module
222           * @param {string} modname Name of the module to be loaded.
223           */
224          loadModule: function loadModule (modname)
225          {
226              if (this.modules[modname] && !(this.modules[modname] instanceof Error)) return;
227 
228 
229              try {
230                  var mod = sys.exec("js_modules/" + modname +".js");
231 
232                  mod.modname = modname;
233 
234                  this.modules[modname] = mod;
235 
236                  if (mod.include) for (var x in mod.include)
237                  {
238                      var temp = sys.exec("js_modules/" + mod.include[x] + ".js");
239 
240                      for (var x2 in temp)
241                      {
242                          if (x2 in mod && mod[x2] != null)
243                          {
244                              if (typeof mod[x2] != typeof temp[x2] || (typeof mod[x2] != "object" && typeof mod[x2] != "function"))
245                              {
246                                  throw new Error("Unable to merge");
247                              }
248                              if (typeof mod[x2] === "function")
249                              {
250                                  // use a closure to merge the two functions as one
251                                  mod[x2] = (function (m, t) {
252                                                 return function ()
253                                                 {
254                                                     m.apply(mod, arguments);
255                                                     t.apply(mod, arguments);
256                                                 };
257                                             })(mod[x2], temp[x2]);
258                              }
259                              else if (mod[x2] instanceof Array)
260                              {
261                                  mod[x2] = mod[x2].concat( temp[x2] );
262                              }
263                              else
264                              {
265                                  for (var x3 in temp[x2])
266                                  {
267                                      if (x3 in mod[x2]) throw new Error("Unable to merge");
268 
269                                      mod[x2][x3] = temp[x2][x3];
270                                  }
271                              }
272                          }
273                          else
274                          {
275                              mod[x2] = temp[x2];
276                          }
277                      }
278                  }
279 
280                  if (!mod.require) mod.require = [];
281                  mod.submodules = [];
282 
283                  for (var x in this.hooks)
284                  {
285                      mod[x] = this.hooks[x];
286                  }
287 
288                  for (var x in mod.require)
289                  {
290                      var reqmodname = mod.require[x];
291 
292                      this.loadModule(reqmodname);
293 
294                      if ( !(reqmodname in this.modules) || this.modules[reqmodname] instanceof Error)
295                      {
296                          this.modules[modname] = new Error("Unmet dependencies");
297                          return;
298                      }
299 
300                      this.modules[reqmodname].submodules.push(modname);
301 
302                      if (this.modules[reqmodname].submodules.indexOf(modname) === -1) throw new Error(":(");
303 
304                      Object.defineProperty(this.modules[modname], reqmodname, {configurable: true, value: this.modules[reqmodname]});
305                  }
306 
307                  Object.defineProperty(this.modules[modname], "script", {configurable : true, value: this});
308 
309                  if ("loadModule" in mod)
310                  {
311                      mod.loadModule();
312                  }
313                  this.log("Loaded module: " + modname);
314              }
315              catch (e)
316              {
317 
318                  delete this.modules[modname];
319                  throw e;
320              }
321 
322          }

324          /** Unloads a module
325           * @param {string} modname Name of module to remove.
326           * @return {string[]} List of all modules that were removed, includes others due to dependencies.
327           */
328          unloadModule: function unloadModule (modname, hot)
329          {
330              if ( !(modname in this.modules)) return [];
331              if (this.modules[modname] instanceof Error) return [modname];
332              this.log("Unloading module: " + modname);
333 
334              try
335              {
336 
337 
338 
339                  var unloads = [modname];
340 
341                  var thisModule = this.modules[modname];
342 
343                  var q = [];
344 
345                  thisModule.submodules.forEach(function (i){ q.push(i); });
346 
347                  for (var x in q)
348                  {
349                      unloads.push(q[x]);
350                      var u = this.unloadModule(q[x]);
351                      unloads = unloads.concat(u);
352 
353                  }
354 
355                  if (thisModule.require) for (x in thisModule.require)
356                  {
357                      if (this.modules[thisModule.require[x]].unloadSubmodule)
358                      {
359                          this.modules[thisModule.require[x]].unloadSubmodule(thisModule, modname);
360                      }
361 
362                      this.modules[thisModule.require[x]].submodules.splice(this.modules[thisModule.require[x]].submodules.indexOf(modname), 1);
363                  }
364 
365                  if ("unloadModuleHooks" in thisModule)
366                  {
367                      var unloadModuleHooks = thisModule.unloadModuleHooks;
368                      for (x in unloadModuleHooks)
369                      {
370                          unloadModuleHooks[x].apply(thisModule, [thisModule]);
371                      }
372 
373                  }
374 
375                  if ("unloadModule" in thisModule) try
376                  {
377                      thisModule.unloadModule();
378                  }
379                  catch (e)
380                  {
381                      this.log("[[ERROR:]]" + e.toString() + "\n" + e.backtracetext);
382                  }
383              } catch (e2)
384              {
385 
386              }
387              finally
388              {
389                  delete this.modules[modname];
390                  return unloads;
391              }
392 
393 
394 
395 
396          }

402          {
403 
404              var test1 = ["print","gc","version","global","sys","SESSION","Qt","script"];
405 
406              var test2 = Object.keys(global);
407 
408              var poisoned = false;
409 
410              sys.enableStrict();
411 
412              this.modules = new Object;
413 
414              if (sys.fileExists("ZSCRIPTS_COPYING")) print(sys.read("ZSCRIPTS_COPYING"));
415              else if (sys.fileExists("ZVXSCRIPTS_COPYING")) print(sys.read("ZVXSCRIPTS_COPYING"));
416              else print("!!! Missing license, see http://github.com/archzombie/zvxscripts !!!");
417 
418              for (var x in test2) if (test1.indexOf(test2[x]) === -1)
419              {
420                  print("WARNING: Global object poisoned. Removing property: " + test2[x]);
421                  delete global[test2[x]];
422                  poisoned = true;
423              }
424 
425              if (poisoned) gc();
426 
427              this.registerHandler("beforeLogIn", this, "AGPL");
428 
429              sys.unsetAllTimers();
430 
431              try {
432                  var f;
433                  if (sys.fileExists("main.json")) f = sys.read("main.json");
434 
435                  else f = "{}";
436 
437                  var o = JSON.parse(f);
438 
439                  if (typeof o == typeof new Object);
440 
441                      else throw new Error("Corrupt File");
442 
443                  this.config = o;
444 
445                  if (!this.config.modules) this.config.modules = ["default"];
446 
447                  for ( x in this.config.modules)
448                  {
449                      this.loadModule(this.config.modules[x]);
450                  }
451 
452                  sys.writeToFile("main.json", JSON.stringify(this.config));
