 22 /*
 23 profile.js
 24 
 25 Implements:
 26 
 27 profileMatches(src):
 28   Returns a list of all the profile IDs the player matches
 29 
 30 profileID(src):
 31   Returns a single ID for the player. If there are multiple matches,
 32   it merges them together. If there are none, it creates a new profile
 33 
 34 profileByName(name):
 35   Returns if any profile matches this name, the profile ID, otherwise -1
 36 
 37 profileByIp(ipaddr):
 38   Returns a matching profile ID, otherwise -1
 39 
 40 trace(profid):
 41   When profiles are merged one profile will take priority, but the others
 42   will still exist
 43 
 44 profileUpdateInfo(prof, src):
 45   Updates profile information
 46 
 47 
 48 */
 49 
 50 
 51 ({
 52     require: ["io", "logs"]
 53     , 
 54     database: null
 55     /* <Object
 56        profiles: <Object Key:[<Int indexForProfile>] Value:[<Object profile>] >
 57        ,
 58        profile_counter: <INT>
 59        > */
 60     ,
 61     relationaldatabase: new Object
 62     /* <Object
 63        names: <Object Key:[<String "$"> + <String name>] Value:[<Int indexForProfile>] >
 64        ,
 65        ips: <Object Key:[<String ipaddr>] Value:[<Int indexForProfile>] >
 66     */
 67     ,
 68     users: new Object
 69     ,
 70     loadModule: function ()
 71     {
 72         this.database = this.io.openDB("profile");
 73 
 74         if (!this.database.profiles) this.database.profiles = new Object;
 75         if (!this.database.profile_counter) this.database.profile_counter = 0;
 76 
 77         var uids = sys.playerIds();
 78         
 79 
 80         this.updateAllRelations(); 
 81 
 82         for (var x in uids)
 83         {
 84             this.registerPlayer(uids[x]);
 85         }
 86 
 87         
 88     }
 89     ,
 90     unloadModule: function ()
 91     {
 92         this.io.closeDB("profile");
 93     }
 94     ,
 95     updateAllRelations: function()
 96     {
 97         this.relationaldatabase = 
 98             {
 99                 names: new Object,
100                 ips: new Object
101             };
102         
103         for (var x in this.database.profiles)
104         {
105             this.updateProfileRelations(x);
106         }
107     }
108     ,
109     profileID: function (src)
110     {
111         var _;
112 
113         if (_ = this.users[src]) return _;
114 
115         else 
116         {
117             return this.users[src] = this.registerPlayer(src);
118         }
119     }
120     ,
121     lastName: function (prof)
122     {
123         return this.database.profiles[prof].lastName;
124     }
125     ,
126     profileNames: function (prof)
127     {
128         return this.database.profiles[prof].names;
129     }
130     ,
131     profileIPs: function (prof)
132     {
133         return this.database.profiles[prof].ips;
134     }
135     ,
136     updateProfileRelations: function (id)
137     {
138         var prof = this.database.profiles[id];
139 
140         if (prof.mergedInto) return;
141 
142         var prof_names = prof.names;
143         var prof_ips = prof.ips;
144         
145         for (var x in prof_names)
146         {
147             if (!("$"+ prof_names[x] in this.relationaldatabase.names)) this.relationaldatabase.names["$"+ prof_names[x]] = id;
148 
149             else if (this.relationaldatabase.names["$"+prof_names[x]] != id) this.logs.logMessage(this.logs.ERROR, "Error condition, multimatch on profile #" + id);
150         }
151         
152         for (var x in prof_ips)
153         {
154             if (!(prof_ips[x] in this.relationaldatabase.ips)) this.relationaldatabase.ips[prof_ips[x]] = id;
155 
156             else if (this.relationaldatabase.ips[prof_ips[x]] != id) this.logs.logMessage(this.logs.ERROR, "Error condition, multimatch on profile #" + id);
157         }
158     }
159     ,
160     profileMatches: function (src)
161     {
162         var sys_name$src = sys.name(src);
163         var name = sys_name$src.toLowerCase();
164         var ip = sys.ip(src);
165         var matches = new Object;
166 
167         if (ip in this.relationaldatabase.ips)
168         {
169             matches[this.relationaldatabase.ips[ip]] = null;
170         }
171 
172         if ("$"+name in this.relationaldatabase.names)
173         {
174             matches[this.relationaldatabase.names["$"+name]] = null;
175         }
176 
177         return Object.keys(matches);
178     }
179     ,
180     profileUpdateInfo: function (profid, src)
181     {
182         var prof = this.database.profiles[profid];
183         var sys_name$src = sys.name(src);
184         var sys_ip$src = sys.ip(src);
185 
186         if (prof.names.indexOf(sys_name$src.toLowerCase()) == -1) prof.names.push(sys_name$src.toLowerCase());
187 
188         if (prof.ips.indexOf(sys_ip$src) == -1) prof.ips.push(sys_ip$src);
189 
190         prof.lastName = sys_name$src;
191         prof.lastIP = sys_ip$src;
192         prof.lastOnline = +new Date;
193 
194         this.updateProfileRelations(profid);
195 
196         return;
197     }
198     ,
199     registerPlayer: function(src)
200     { 
201 
202         var matchesList = this.profileMatches(src);
203 
204         if (matchesList.length == 0)
205         {
206             var p = this.newProfile(src);
207             this.profileUpdateInfo(p, src);
208             return p;
209         }
210         
211         else if (matchesList.length > 1)
212         {
213             this.logs.logMessage(this.logs.INFO, "Merging profiles " + JSON.stringify(matchesList));
214             this.mergeProfiles(matchesList);
215         }
216         
217         var i = parseInt(matchesList[0]); 
218         var prof = this.database.profiles[i];   
219 
220         this.profileUpdateInfo(i, src);
221 
222         return i;
223     }
224     ,
225     profileByIP: function (ip)
226     {
227         if (ip in this.relationaldatabase.ips) return this.relationaldatabase.ips[ip];
228 
229         else return -1;
230     }
231     ,
232     profileByName: function (n)
233     {
234         var name = "$"+ n.toLowerCase();
235         if (name in this.relationaldatabase.names) return this.relationaldatabase.names[name];
236 
237         else return -1;        
238     }
239     ,
240     newProfile: function (src)
241     {
242         var prof = new Object;
243         var prof_id = this.database.profile_counter++;
244 
245         prof.names = [];
246         prof.ips = [];
247         this.database.profiles[prof_id] = prof;
248 
249         return prof_id;
250     }
251     ,
252     trace: function (prof)
253     {
254         var p = this.database.profiles[prof];
255 
256         var idx = prof;
257         while (p.mergedInto)
258         {
259             idx = p.mergedInto;
260             p = this.database.profiles[p.mergedInto];
261         }
262 
263         return idx;
264     }
265     ,
266     mergeProfiles: function (list)
267     {
268         var origin = this.database.profiles[list[0]];
269 
270         this.logs.logMessage(this.logs.WARN, "Merging profiles " + JSON.stringify(list));
271 
272         for (var x1 in list)
273         {
274             if (x1 == 0) continue;
275 
276             this.database.profiles[list[x1]].mergedInto = list[0];
277             for (var x2 in this.database.profiles[list[x1]].names)
278             {
279                 if (origin.names.indexOf(this.database.profiles[list[x1]].names[x2]) == -1)
280                 {
281                     origin.names.push(this.database.profiles[list[x1]].names[x2]);
282                 }
283             }
284 
285             for (var x2 in this.database.profiles[list[x1]].ips)
286             {
287                 if (origin.ips.indexOf(this.database.profiles[list[x1]].ips[x2]) == -1)
288                 {
289                     origin.ips.push(this.database.profiles[list[x1]].ips[x2]);
290                 }                
291             }
292             
293         }
294     }
295 
296 });
