 22 ({
 23     entityTick: function (e)
 24     {
 25 
 26         e.sta += e.maxsta/300;
 27         e.msta += e.maxmsta/60;
 28 
 29         // Stamina
 30         if (e.sp < 0)
 31             // No stamina
 32         {
 33             e.hp += e.sp;
 34             // Reduce the HP by the amount of missing stamina.
 35 
 36             e.sta = 0;
 37             // Reset stamina
 38         }
 39         if (e.sp > e.maxsp / 3)
 40             // Has a fairly large amount of stamina
 41         {
 42             hp += maxhp/600;
 43             // Recover some HP
 44         }
 45         if (e.sp > e.maxsp)
 46             // Too much stamina!
 47         {
 48             e.sp = e.maxsp;
 49             // Remove overflow
 50             
 51             hp += maxhp/300;
 52             // recover extra hp!
 53         }
 54 
 55         // Mental Stamina
 56         if (e.msta < 0)
 57             // No mental stamina
 58         {
 59             e.mp -= e.msta;
 60             e.hp -= e.msta;
 61             // Lose hp and mana
 62 
 63             e.msta = 0;
 64             // Reset
 65         }
 66         if (e.msta > e.maxmsta / 3)
 67             // Recover mana
 68         {
 69             e.mana += e.maxmana/100;
 70             // increase
 71         }
 72 
 73         // Mana
 74         if (e.mana < 0)
 75             // Not enough mana
 76         {
 77             e.sta += e.mana;
 78             e.msta += e.mana;
 79             // Decrease staminas
 80 
 81             e.mana = 0;
 82             // Reset mana;
 83         }
 84         if (e.mana > e.maxmana)
 85             // Too much mana
 86         {
 87             if (e.mana > e.maxmana *1.1) e.hp += e.maxmana*1.1 - e.mana;
 88             // Decrease HP by overflow, uncontrolable mana damages body!
 89             
 90             e.mana = e.maxmana;
 91             // Reset
 92         }
 93 
 94         // Health
 95         if (e.hp <= 0 && !e.invincible && (!e.undead && e.hp > -e.maxhp*5))
 96             // check for death
 97         {
 98             this.entityDie(e);
 99         }
100         else if (e.hp > e.maxhp)
101             // overflow
102         {
103             e.hp = e.maxhp;
104             // reset
105         }
106         
107 
108         
109     }
110     
111 });
