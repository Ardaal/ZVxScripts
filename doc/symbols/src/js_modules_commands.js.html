 40 /** Exectutes the command, ignoring permission. (best to check for permission first)
 41  * @name code
 42  * @memberOf commandDescriptor.prototype
 43  * @function
 44  * @param {Number} src User ID
 45  * @param {Object} cmd Parsed command
 46  * @param {Number} chan Channel ID
 47  * @param {Object} [cache] Cache object as a result of perm function if applicable. Must not assume this is present.
 48  */
 49 /** Describes the command
 50  * @name desc
 51  * @memberOf commandDescriptor.prototype
 52  * @type String
 53  */
 54 /** If the server console can use the command, requires special handling in some cases.
 55  * @name server
 56  * @memberOf commandDescriptor.prototype
 57  * @type Boolean
 58  */
 59 /** Describes the command's options. Key is option name, value is description.
 60  * @name options
 61  * @memberOf commandDescriptor.prototype
 62  * @type {Object.<String>}
 63  */
 64 /** @scope script.modules.commands */
 65 ({
 66      require: ["com", "theme", "parsecommand", "util", "logs", "io", "dmp", "user"],
 67 
 68 
 69      dmpO: null,
 70 
 71 
 72      /** A list of all the command object descriptors loaded
 73       * @type {Object.<commandDescriptor>}
 74       */
 75      commands_db: new Object,
 76 
 77 
 78      /** Error of Permission. */
 79      PERMISSION_ERROR: new Object,
 80      FORMAT_ERROT: new Object,
 81 
 82      /** @event */
 83      loadModule: function ()
 84      {
 85 
 86          this.dmpO = new this.dmp.constructor();
 87 
 88          this.dmpO.Match_Threshold = 0.5;
 89 
 90          this.dmpO.Match_Distance = 0;
 91 
 92          //this.io.registerConfig(this, { ownerHasAllCommands: false });
 93 
 94          //if (!this.config.commands) this.config.commands = new Object;
 95 
 96 	 this.user.registerConfigHook(this, "userConfiguration");
 97      },
 98 
 99 
100      userConfiguration: function (conf)
101      {
102          if (!conf.commandParser || !(conf.commandParser in this.parsecommand.commandParsers)) conf.commandParser = "optargs";
103      },
104 
105 
106      /** Registeres command.
107       * @param {String} name Name of command.
108       * @param {Module} object Module object.
109       * @param {String} [prop=name] Name of property from module to use
110       */
111      registerCommand: function (name, object, prop)
112      {
113          if (name in this.commands_db)
114          {
115              this.script.log("WARN: Overwriting command " +name);
116          }
117 
118          var comnd = object[prop || name];
119          comnd.bind = object;
120          comnd.name = name;
121 
122          this.commands_db[name] = object[prop || name];
123          //this.commands_db[name].config = this.config.commands[name];
124 
125          if (comnd.aliases) for (var x in comnd.aliases)
126          {
127              this.commands_db[comnd.aliases[x]] = comnd;
128          }
129 
130          object.onUnloadModule(
131              this.util.bind(
132                  this,
133                  function ()
134                  {
135                      this.unregisterCommand(name);
136                  }
137              )
138          );
139          return;
140      },
141 
142 
143 
144      /** Unregisters a command
145       * @param {String} name Name of command to unregister
146       * */
147      unregisterCommand: function (name)
148      {
149          if (this.commands_db[name])
150          {
151              if (this.commands_db[name].aliases) for (var x in this.commands_db[name].aliases)
152              {
153                  delete this.commands_db[this.commands_db[name].aliases[x]];
154              }
155          }
156          delete this.commands_db[name];
157      },
158 
159 
160 
161      serverCanUseCmd: function (name)
162      {
163          return this.commands_db[name].server;
164      },
165 
166 
167 
168      /** Checks if player has permission to use a command
169       * @param {Number} src Player ID.
170       * @param {parsedCommand} cmd
171       * @param {Number} chan Channel ID.
172       */
173      commandPerm: function (src, cmd, chan)
174      {
175          var cmdobj = this.commands_db[cmd.name];
176 
177          if (src == 0) return cmdobj.server;
178 
179          if (this.user.hasPerm(src, "COMMAND[" + cmdobj.name.toUpperCase() + "]")) return true;
180 
181          else if (typeof cmdobj.perm == "string")
182          {
183              return this.user.hasPerm(src, cmdobj.perm);
184          }
185          else if (typeof cmdobj.perm == "boolean")
186          {
187              return cmdobj.perm;
188          }
189          else if (cmdobj.perm === undefined)
190          {
191              return false;
192          }
193          else
194          {
195 
196              return (cmdobj.perm || cmdobj.perm2).call(cmdobj.bind, src, cmd, chan);
197          }
198      },
199 
200 
201      tryCommand: function(src, cmd, chan)
202      {
203          var cmd_obj = this.commands_db[cmd.name];
204 
205          if (!cmd_obj)
206          {
207              this.com.message([src], "Command does not exist.", this.theme.WARN);
208 
209              if (1) // bugfix
210              {
211                  var matches = [];
212 
213                  for (var x in this.commands_db) if (this.dmpO.match_main(x, cmd.name, 0) != -1)
214                  {
215                      matches.push(x);
216                  }
217 
218                  if (matches.length) this.com.message(src, "Did you mean one of these?: " + matches.join(", ") + "?");
219 
220              }
221 
222              return;
223          }
224 
225          var perm = this.commandPerm(src, cmd, chan);
226 
227          if (!perm)
228          {
229              this.com.message(src, "Permission denied.", this.theme.WARN);
230              return;
231          }
232 
233          try
234          {
235              cmd_obj.code.call(cmd_obj.bind, src, cmd, chan, (perm === true ? void 0 : perm));
236          }
237          catch (e)
238          {
239              this.logs.logMessage(this.logs.ERROR, "Caught error in " + e.fileName + " at line #" + e.lineNumber + ": " + e.toString(), e.backtracetext);
240              this.com.broadcast("Script Error, check logs.", this.theme.CRITICAL);
241          }
242      },
243 
244 
245      /** Parses text from a user as a command, checks relevant permissions etc.
246       * @event
247       * @param src User ID
248       * @param text Unparsed text object
249       * @param chan Channel ID
250       */
251      issueCommand: function(src, text, chan)
252      {
253          var cmd = this.parsecommand.commandParsers[this.user.userConfig(src).commandParser](text);
254 
255          this.logs.logMessage(this.logs.COMMAND, (chan == -1 ? "[N/A] " : "[#"+sys.channel(chan)+"] ") + this.user.name(src) + ": " + text);
256 
257          this.tryCommand(src, cmd, chan);
258      }
259  });
