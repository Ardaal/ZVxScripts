 22 /** Commands
 23  * @name commands
 24  * @memberOf script.modules
 25  * @namespace
 26  * */
 27 /**
 28  * @name commandDescriptor
 29  * @class
 30  */
 31 /** Check for permission.
 32  * @name perm
 33  * @memberOf commandDescriptor.prototype
 34  * @function
 35  * @param {Number} src User ID
 36  * @param {Object} cmd Parsed command
 37  * @param {Number} chan Channel ID
 38  * @return {Boolean|Object} True/False, or if true, may also be object with cache.
 39  */
 40 /** Exectutes the command, ignoring permission. (best to check for permission first)
 41  * @name code
 42  * @memberOf commandDescriptor.prototype
 43  * @function
 44  * @param {Number} src User ID
 45  * @param {Object} cmd Parsed command
 46  * @param {Number} chan Channel ID
 47  * @param {Object} [cache] Cache object as a result of perm function if applicable. Must not assume this is present.
 48  */
 49 /** Describes the command
 50  * @name desc
 51  * @memberOf commandDescriptor.prototype
 52  * @type String
 53  */
 54 /** Describes the command's options. Key is option name, value is description.
 55  * @name options
 56  * @memberOf commandDescriptor.prototype
 57  * @type {Object.<String>}
 58  */
 59 /** @scope script.modules.commands */
 60 ({
 61     require: ["com", "theme", "parsecommand", "util", "logs", "io"]
 62     ,
 63     /** A list of all the command object descriptors loaded
 64      * @type {Object.<commandDescriptor>}
 65      */
 66     commands_db: new Object
 67     ,
 68     /** @event */
 69     loadModule: function ()
 70     {
 71 
 72     }
 73     ,
 74     /** Registeres command.
 75      * @param {String} name Name of command.
 76      * @param {Module} object Module object.
 77      * @param {String} [prop=name] Name of property from module to use
 78      */
 79     registerCommand: function (name, object, prop)
 80     {
 81         if (name in this.commands_db)
 82         {
 83             this.script.log("WARN: Overwriting command " +name);
 84         }
 85 
 86         var comnd = object[prop || name];
 87         comnd.bind = object;
 88         comnd.name = name;
 89 
 90         var cfg = this.io.readConfig(name + "_command", {specialUsers:{}});
 91 
 92         this.commands_db[name] = object[prop || name];
 93         this.commands_db[name].config = cfg;
 94 
 95         if (comnd.aliases) for (var x in comnd.aliases)
 96         {
 97             this.commands_db[comnd.aliases[x]] = comnd;
 98         }
 99 
100         object.onUnloadModule( this.util.bind(
101             this,
102             function ()
103             {
104                 this.unregisterCommand(name);
105             }
106         ));
107         return;
108     }
109     ,
110     /** Unregisters a command
111      * @param {String} name Name of command to unregister
112      * */
113     unregisterCommand: function (name)
114     {
115         if (this.commands_db[name])
116         {
117             if (this.commands_db[name].aliases) for (var x in this.commands_db[name].aliases)
118             {
119                 delete this.commands_db[this.commands_db[name].aliases[x]];
120             }
121         }
122         delete this.commands_db[name];
123     }
124     ,
125     /** Checks if player has permission to use a command
126      * @param {Number} src Player ID.
127      * @param {parsedCommand} cmd
128      * @param {Number} chan Channel ID.
129      */
130     commandPerm: function (src, cmd, chan)
131     {
132         var cmdobj = this.commands_db[cmd.name];
133         if (sys.auth(src) == 3) return true;
134 
135         else if (cmdobj.config.specialUsers[sys.name(src).toLowerCase()])
136         {
137             return true;
138         }
139 
140         else
141         {
142             return (cmdobj.perm || cmdobj.perm2).call(cmdobj.bind, src, cmd, chan);
143         }
144     }
145     ,
146     /** Parses text from a user as a command, checks relevant permissions etc.
147      * @event
148      * @param src User ID
149      * @param text Unparsed text object
150      * @param chan Channel ID
151      */
152     issueCommand: function(src, text, chan)
153     {
154         var cmd = this.parsecommand.parseCommand(text);
155 
156         var cmd_obj = this.commands_db[cmd.name];
157 
158         if (!cmd_obj)
159         {
160             this.com.message([src], "Command does not exist.", this.theme.WARN);
161             return;
162         }
163 
164         var perm = this.commandPerm(src, cmd, chan);
165         if (!perm)
166         {
167             this.com.message([src], "Permission denied.", this.theme.WARN);
168             return;
169         }
170 
171         try
172         {
173             cmd_obj.code.call(cmd_obj.bind, src, cmd, chan, (perm === true ? void 0 : perm));
174         }
175         catch (e)
176         {
177             this.logs.logMessage(this.logs.ERROR, "Caught error in " + e.fileName + " at line #" + e.lineNumber + ": " + e.toString() + "\n" + e.backtracetext);
178             this.com.broadcast("Script Error, check logs.", this.theme.CRITICAL);
179         }
180     }
181 });
