 22 /** Command parser
 23  * @name parsecommand
 24  * @memberOf script.modules
 25  * @namespace
 26  * */
 27 /** A parsed command
 28  * @name parsedCommand
 29  * @class
 30  */
 31 /** A list of all the arguments
 32  * @name args
 33  * @memberOf parsedCommand.prototype
 34  * @type Array
 35  */
 36 /** A list of all the flags
 37  * @name flags
 38  * @memberOf parsedCommand.prototype
 39  * @type Object
 40  */
 41 /** @scope script.modules.parsecommand */
 42 ({
 43     require: []
 44     ,
 45     /** Parses a command.
 46      * @param {String} text The text to parse.
 47      * @return {parsedCommand} The parsed command.
 48      * */
 49 
 50     parseCommand: function (text)
 51     {
 52         var cmd = new Object;
 53 
 54         var match = text.match(/^(?:!|\/)([^\s]*)(?:\s+(.*))?/i);
 55 
 56         cmd.args = [];
 57         cmd.flags = {};
 58 
 59         if (!match) return cmd;
 60 
 61         cmd.name = match[1];
 62         cmd.input = match[2];
 63 
 64         var input = cmd.input;
 65 
 66         if (!input) return cmd;
 67 
 68         while (input != (input =
 69                          input.replace(/^\s*(?:\-{1,2}((?:\S|\\ )+)(?:\=(?:\"((?:\\.|[^\"])+)\"|((?:\w|\\ )+)))?)|(?:(?:\"((?:\\.|[^\"])+)\")|((?:\w|\\ )+))/, cl_next))
 70                //                                      ^flagname              ^flagvalstr        ^flagvalueraw             ^argvalstr         ^argvalraw
 71               ) {};
 72 
 73         function cl_next (text, flagname, flagvaluestr, flagvalueraw, argvaluestr, argvalueraw)
 74         {
 75             if (flagname)
 76             {
 77                 var f = ((flagvaluestr ? flagvaluestr: void 0) || flagvalueraw);
 78 
 79                 if (f)
 80                 {
 81                     cmd.flags[flagname] = f.replace(/\\(.)/g, "$1");
 82                 }
 83                 else
 84                 {
 85                     cmd.flags[flagname] = true;
 86                 }
 87             }
 88             else
 89             {
 90                 cmd.args.push(((argvaluestr !== undefined ? argvaluestr: void 0) || argvalueraw).replace(/\\(.)/g, "$1"));
 91             }
 92             return "";
 93         }
 94 
 95         return cmd;
 96 
 97     }
 98 
 99 });
