 94      /** Inspects variant
 95       * @param variant Value to be inspected
 96       * @param {Boolean} exclude If duiplicates should only be printed once. (recommended)
 97       * @return {String} Human readable representation of variant.
 98       */
 99      inspect: function (variant, exclude)
100      {
101          // Rather complicated function, bear with it!
102          /** The stack frames variable includes all the objects we are already inside of, to catch circular references. */
103          var stack_frames = [];
104 
105          /** The refs varible keeps track of all the objects we have touched at any time, to catch duiplcate references */
106          var refs = [];
107 
108 
109          function serialize (variant)
110          {
111              if (typeof variant === "undefined")
112              {
113                  return "<Undefined>";
114              }
115 
116              if (variant === null)
117              {
118                  return "<Null>";
119              }
120 
121              if (typeof variant === "number")
122              {
123                  return "<Number> " + variant;
124              }
125 
126              if (typeof variant === "string")
127              {
128                  return "<String> " + JSON.stringify(variant);
129              }
130 
131              if (typeof variant === "boolean")
132              {
133                  return "<Bool> " + JSON.stringify(variant);
134              }
135 
136              if (typeof variant === "function")
137              {
138                  // Right now most all fucntions would be anonymous, but later we may add features to sys to make it able to read more function information :)
139                  return "<Function> [[" + (variant.name?variant.name:"<anonymous>") + "]]";
140              }
141 
142              if (typeof variant === "object")
143              {
144                  var refidx = refs.indexOf(variant);
145                  var dup = true;
146 
147                  var str = "";
148 
149                  if (refidx === -1)
150                  {
151                      refidx = refs.length;
152                      refs.push(variant);
153                      dup = false;
154                  }
155 
156                  str += "<" + (variant instanceof Array ? "Array" : "Object") + " &" + refidx + ">";
157                  // Arrays still do not look very nice, they have object syntax.
158 
159 
160                  if (stack_frames.indexOf(variant) != -1)
161                  {
162                      str += " [[Circular Reference]]";
163                      return str;
164                  }
165                  else if (dup)
166                  {
167                      str += " [[Duplicate Reference]]";
168                      return str;
169                  }
170 
171                  str += "\n";
172 
173                  for (var i = 0; i < stack_frames.length; i++) str += "    ";
174 
175                  str += "{\n";
176 
177                  stack_frames.push(variant);
178 
179                  for (var x in variant)
180                  {
181                      for (i = 0; i < stack_frames.length; i++) str += "    ";
182                      str += JSON.stringify(x) + ": " + serialize(variant[x]) + "\n";
183                  }
184 
185                  stack_frames.pop();
186 
187                  for (i = 0; i < stack_frames.length; i++) str += "    ";
188                  str += "}";
189 
190                  return str;
191              }
192 
193              return "?";
194          }
195 
196          return serialize(variant);
197      },

205      /** Generates a registor for use by the module.
206       * @param host The module to use.
207       * @param {this.LIST_REGISTOR|this.ASSOCIATIVE_REGISTOR} type
208       * @param {String} storename The name of the property on the host to use as the registry.
209       * @param {Boolean} direct If the registor copies the value directly or from a property of the registering module.
210       * @param {String} onRegister Name of callback event to use when the program is run.
211       */
212      generateRegistor: function (host, type, storename, direct, onRegister)
213      {
214          if (type === this.LIST_REGISTOR)
215          {
216              host[storename] || (host[storename] = []);
217 
218              return function (module, applicant)
219              {
220                  var registrant = (direct? applicant : module[applicant]);
221 
222                  if (typeof registrant === "function" || typeof registrant === "object")
223                  {
224                      Object.defineProperty(registrant, "module", {value: module, configurable: true});
225                  }
226 
227                  if (typeof registrant === "function")
228                  {
229                      registrant = this.util.bind(module, registrant);
230                  }
231 
232                  host[storename].push(registrant);
233 
234                  module.onUnloadModule(
235                      this.util.bind(
236                          this,
237                          function ()
238                          {
239                              host[storename].splice(host[storename].indexOf(registrant), 1);
240                          }
241                      )
242                  );
243 
244                  if (host[onRegister]) return host[onRegister](module, registrant);
245 
246                  else return void 0;
247 
248              };
249 
250          }
251          else if (this.ASSOCIATIVE_REGISTOR === type) return function (module, applicant, propname)
252          {
253              host[storename] || (host[storename] = new Object);
254 
255              var registrant = module[propnamme||applicant];
256 
257              if (typeof registrant === "function" || typeof registrant === "object")
258              {
259                  Object.defineProperty(registrant, "module", {value: module, configurable: true});
260              }
261 
262              if (typeof registrant === "function")
263              {
264                  registrant = this.util.bind(module, registrant);
265              }
266 
267              host[storename][applicant] = registrant;
268 
269              module.onUnloadModule(
270                  function ()
271                  {
272                      host[storename].splice(host[storename].indexOf(registrant), 1);
273                  }
274              );
275 
276 
277          };
278 
279 
280          else return null; // unimplemented

296      intToBin: function (i)
297      {
298          if (this.cache[i]) return this.cache[i];
299          var bin = new sys.ByteArray(4);
300 
301 
302          bin[0] = i & 255;
303          if (i > 255)
304              bin[1] = (i >> 8) & 255;
305          if (i > 65535)
306              bin[2] = (i >> 16) & 255;
307          if (i > 16777215)
308              bin[3] = (i >> 24) & 255;
309 
310          return this.cache[i] = bin;
311      },

323      ipMatchesSubnet: function (ip, subnet)
324      {
325          ip = ip.split(/\./g);
326 
327          var subnet_ip = subnet.split(/\//)[0].split(/\./g);
328          var subnet_mask =  subnet.split(/\//)[1];
329 
330          var ipAddr = this.binToIntO(ip);
331          var subAddr = this.binToIntO(subnet_ip);
332 
333          if (subnet_mask == 0) return true;
334 
335          ipAddr &= ~((1 << (32 - +subnet_mask)) - 1);
336 
337          return ipAddr == subAddr;
