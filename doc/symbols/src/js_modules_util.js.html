 22 /** Utilities module
 23  * @memberof script.modules
 24  * @name util
 25  * @namespace
 26  */
 27 /** @scope script.modules.util */
 28 ({
 29     /** Binds a function and object together.
 30      * @param {Object} obj The object to be used as "this" when calling func.
 31      * @param {Function} func The function to be called.
 32      * @return A new function which is a clone of func, but where the this object is always obj.
 33      */
 34     bind: function (obj, func)
 35     {
 36         return function ()
 37         {
 38             func.apply(obj, arguments);
 39         };
 40 
 41     }
 42     ,
 43     /** Converts a value to an array if it isn't already one.
 44      * @param variant The value to be wrapped in an array if it isn't already one
 45      * @return {Array}
 46      * @example arrayify(8) >> [8], arrayify([7, 8]) >> [7, 8]
 47      */
 48     arrayify: function (variant)
 49     {
 50         if (typeof variant !== "object" || !(variant instanceof Array)) return [variant];
 51 
 52         return variant;
 53     }
 54     ,
 55     /** Shuffles the array.
 56      * @param array Array to be shuffled.
 57      */
 58     shuffle: function (array)
 59     {
 60         // fisheryates
 61         var i = array.length;
 62         var temp;
 63         var rand;
 64 
 65         for (i = array.length; i;)
 66             // Shuffles the array
 67         {
 68             rand = Math.floor(Math.random() * i--);
 69             temp = array[i];
 70             array[i] = array[rand];
 71             array[rand] = temp;
 72         }
 73     }
 74     ,
 75     /** Inspects variant
 76      * @param variant Value to be inspected
 77      * @param {Boolean} exclude If duiplicates should only be printed once. (recommended)
 78      * @return {String} Human readable representation of variant.
 79      */
 80     inspect: function (variant, exclude)
 81     {
 82         // Rather complicated function, bear with it!
 83         /** The stack frames variable includes all the objects we are already inside of, to catch circular references. */
 84         var stack_frames = [];
 85 
 86         /** The refs varible keeps track of all the objects we have touched at any time, to catch duiplcate references */
 87         var refs = [];
 88 
 89 
 90         function serialize (variant)
 91         {
 92             if (typeof variant === "undefined")
 93             {
 94                 return "<Undefined>";
 95             }
 96 
 97             if (variant === null)
 98             {
 99                 return "<Null>";
100             }
101 
102             if (typeof variant === "number")
103             {
104                 return "<Number> " + variant;
105             }
106 
107             if (typeof variant === "string")
108             {
109                 return "<String> " + JSON.stringify(variant);
110             }
111 
112             if (typeof variant === "function")
113             {
114                 // Right now most all fucntions would be anonymous, but later we may add features to sys to make it able to read more function information :)
115                 return "<Function> [[" + (variant.name?variant.name:"<anonymous>") + "]]";
116             }
117 
118             if (typeof variant === "object")
119             {
120                 var refidx = refs.indexOf(variant);
121                 var dup = true;
122 
123                 var str = "";
124 
125                 if (refidx === -1)
126                 {
127                     refidx = refs.length;
128                     refs.push(variant);
129                     dup = false;
130                 }
131 
132                 str += "<" + (variant instanceof Array ? "Array" : "Object") + " &" + refidx + ">";
133                 // Arrays still do not look very nice, they have object syntax.
134 
135 
136                 if (stack_frames.indexOf(variant) != -1)
137                 {
138                     str += " [[Circular Reference]]";
139                     return str;
140                 }
141                 else if (dup)
142                 {
143                     str += " [[Duplicate Reference]]";
144                     return str;
145                 }
146 
147                 str += "\n";
148 
149                 for (var i = 0; i < stack_frames.length; i++) str += "    ";
150 
151                 str += "{\n";
152 
153                 stack_frames.push(variant);
154 
155                 for (var x in variant)
156                 {
157                     for (i = 0; i < stack_frames.length; i++) str += "    ";
158                     str += JSON.stringify(x) + ": " + serialize(variant[x]) + "\n";
159                 }
160 
161                 stack_frames.pop();
162 
163                 for (i = 0; i < stack_frames.length; i++) str += "    ";
164                 str += "}";
165 
166                 return str;
167             }
168 
169             return "?";
170         }
171 
172         return serialize(variant);
173     }
174 });
