 63      loadModule: function ()

 79      /** Registers the module's module.config property.
 80       * @param module The "this" modules to apply the function to.
 81       * @param defs The default configuration
 82       */
 83      registerConfig: function (module, defs)
 84      {
 85          if (!this.configs[module.modname]) this.configs[module.modname] = this.openDB(module.modname + ".config");
 86 
 87          for (var x in defs)
 88          {
 89              if (! (x in this.configs[module.modname]) ) this.configs[module.modname][x] = defs[x];
 90          }
 91 
 92          module.config = this.configs[module.modname];

178      /** Opens a database
179       * @returns {IOdatabase}
180       */
181      openDB: function (dbname, dboptions)
182      {
183          if (! dboptions) dboptions = new Object;
184          var start = +new Date;
185          var end;
186          if (dbname in this.openDBs)
187          {
188              return this.openDBs[dbname].db;
189          }
190          var db, patches, dbo, dataText;
191 
192          get_data:
193          {
194              if (!sys.fileExists("js_databases/" + dbname + ".jsqz"))
195              {
196                  db = new Object;
197                  sys.writeObject("js_databases/" +dbname + ".jsqz", db, 9);
198                  break get_data;
199              }
200              else
201              {
202                  db = sys.readObject("js_databases/" + dbname + ".jsqz");
203              }
204 
205              if (!sys.fileExists("js_databases/" + dbname + ".jsqz.transactions")) break get_data;
206 
207              this.logs.logMessage(this.logs.IO, "Applying patches to database " + dbname);
208 
209              dataText = this.zsrx.zsrx(db, null, 1);
210 
211              patches = sys.read("js_databases/" + dbname + ".jsqz.transactions").split(/\n/g);
212 
213              if (patches[patches.length - 1] === "") patches.pop();
214 
215              try
216              {
217                  for (var x in patches)
218                  {
219                      var parsed = JSON.parse(patches[x]);
220                      {
221                          dataText = this.dmp.def.patch_apply(parsed, dataText)[0];
222                      }
223                  }
224              }
225              catch(e)
226              {
227                  print(JSON.stringify(patches));
228                  print(dataText);
229                  throw e;
230              }
231 
232 
233 	     try
234              {
235                  db = JSON.parse(dataText);
236              }
237              catch(e)
238              {
239                  print(dataText);
240                  throw e;
241              }
242 
243              sys.writeObject("js_databases/" +  dbname + ".jsqz", db, 3);
244 
245              sys.rm("js_databases/" + dbname + ".jsqz.transactions");
246          }
247 
248          if (!dataText) dataText = this.zsrx.zsrx(db);
249 
250          dbo = { db: db, lastSave: +new Date, lastCommit: +new Date, dataText: dataText, hasChanges:null };
251 
252          this.openDBs[dbname] = dbo;
253 
254          end = +new Date;
255 
256 
257          for (var x in this.IOWatchers) try {
258              this.IOWatchers[x](dbname, this.OPEN, (end-start));
259          } catch (_) {}
260          // this.logs.logMessage(this.logs.IO, "Opened database " + dbname + ", took " + (end - start) + "ms.");
261 
262          return db;
263      }

265      /** Writes all changes to file */
266      flushDB: function (dbname)
267      {
268          var metadb = this.openDBs[dbname];
269          var db = metadb.db;
270          var start = +new Date;
271 
272          sys.writeObject("js_databases/" + dbname + ".jsqz", db, 3);
273          if (metadb.hasChanges === true) metadb.hasChanges = false;
274          metadb.dataText = JSON.stringify(db, null, 1);
275          if (sys.fileExists("js_databases/" + dbname + ".jsqz.transactions")) sys.rm("js_databases/" + dbname + ".jsqz.transactions");
276 
277          metadb.lastSave = +new Date;
278          var end = +new Date;
279          for (var x in this.IOWatchers) try {
280              this.IOWatchers[x](dbname, this.SYNC, (end-start));
281          } catch (_) {}
282      }
283      ,
284      /** Commits all changes to file, this may be slower or faster than flush depending on the computer. Fast CPU: Use commit; Fast disk: Use flush. */
285      commitDB: function (dbname)
286      {
287 
288          var start = +new Date;
289 
290          var newData = this.zsrx.zsrx(this.openDBs[dbname].db);
291 
292          if (newData === this.openDBs[dbname].dataText) return;
293 
294          var patch = this.dmp.def.patch_make(this.openDBs[dbname].dataText, this.dmp.def.diff_lineMode_(this.openDBs[dbname].dataText, newData));
295 
296          sys.append("js_databases/" + dbname + ".jsqz.transactions", this.zsrx.zsrx(patch).replace(/\n/g, " ") + "\n");
297 
298          this.openDBs[dbname].dataText = newData;
299 
300          var end = +new Date;
301 
302          for (var x in this.IOWatchers) try {
303              this.IOWatchers[x](dbname, this.COMMIT, (end-start));
304          } catch (e) { this.logs.logMessage(this.logs.ERROR, "I/O Watcher error: " + e.toString(), e.backtracetext); }
305      },
306 
307      /** Marks a database as changed */
308      markDB: function (dbname)
309      {
310          this.openDBs[dbname].hasChanges = true;
311          for (var x in this.IOWatchers) try {
312              this.IOWatchers[x](dbname, this.MARK, 1);
313          } catch (_) {}
314      },
315 
316      /** Closes an open database */
317      closeDB: function (dbname)
318      {
319          var metadb = this.openDBs[dbname];
320          var db = metadb.db;
321 
322          if (!db) return;
323          var start = +new Date;
324          sys.writeObject("js_databases/" + dbname + ".jsqz", db, 8);
325 
326          if (sys.fileExists("js_databases/" +dbname + ".jsqz.transactions")) sys.rm("js_databases/" + dbname + ".jsqz.transactions");
327 
328          delete this.openDBs[dbname];
329          var end = +new Date;
330 
331          for (var x in this.IOWatchers) try {
332              this.IOWatchers[x](dbname, this.CLOSE, (end-start));
333          } catch (_) {}
334 
335      },
336 
337      /** Erase database */
338      purgeDB: function (dbname)
339      {
340 
341 
342          this.logs.logMessage(this.logs.IO, "Purging DB " + dbname);
343 
344          if (this.openDBs[dbname]) this.openDBs[dbname].db = new Object;
345 
346          if (sys.exists("js_databases/" + dbname + ".jsqz")) sys.rm("js_databases/" + dbname + ".jsqz");
347          if (sys.exists("js_databases/" + dbname + ".jsqz.transactions")) sys.rm("js_databases/" + dbname + ".jsqz.transactions");
348 
349      }
350      ,
351      backupDB: function (dbname)
352      {
353          var metadb = this.openDBs[dbname];
354          var db = metadb.db;
355          var start = +new Date;
356          var backupname = dbname + ".backup."+start+".jsqz.bak";
357          sys.writeObject("js_databases/" + backupname, db, 9);
358          var end = +new Date;
359          for (var x in this.IOWatchers) try {
360              this.IOWatchers[x](dbname, this.BACKUP, (end-start));
361          } catch (_) {}
362      }
363      ,
364      step: function ()
365      {
366          var now = +new Date;
367          if (this.config.autosave) for (var x in this.openDBs)
368          {
369              if (this.openDBs[x].lastSave + this.config.autosave <= now)
370              {
371                  var start = +new Date;
372                  if (this.config.autosavemethod === "commit")
373                  {
374                      this.commitDB(x);
375                  }
376                  else
377                  {
378                      this.flushDB(x);
379                  }
380                  var end = +new Date;
381 
382                  this.openDBs[x].lastSave = +new Date;
383 
384                  return;
385              }
386          }
387      }
